031    import bsh.*;

033    /**
034     * The <code>Plex</code> class is the entry point for Plex users into Beanshell.
035     * This file is a trivial modification of the analogous code in Beanshell.
036     *
037     * @version $Id: Plex.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
038     */
039    public class Plex {
040            public static void main( String args[] ) {
041                    if (!Capabilities.classExists( "bsh.util.Util" ))
042                            System.out.println("Can't find the BeanShell utilities.");
043    
044                    if (Capabilities.haveSwing()) {
045                            try {
046                                    new Interpreter().eval("desktop()");
047                            } catch ( EvalError e ) {
048                                    System.err.println("Couldn't start desktop: "+e);
049                            }
050                    } else {
051                            System.err.println("Can't find required javax.swing package");
052                    }
053            }
054    }
