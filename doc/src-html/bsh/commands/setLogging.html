026    package bsh.commands;
027    
028    import java.io.*;
029    import bsh.*;
030    import bsh.PlexReader;
031    
032    /**
033            Set the global defaults for input logging. The default directory is 
034      ~/plex/ and the default setting is off.       
035            @method void setLogging( [ String default_dir ] )
036            @method void setLogging( [ boolean default_on_flag ] )
037    */
038    public class setLogging 
039    {
040            public static String usage() {
041                    return "usage: setLogging( boolean default_on_flag )\n" +
042          "       setLogging( String default_logdir )\n" ;
043            }
044    
045            /**
046         Implement setLogging(String default_dir) command.
047            */
048            public static void invoke(Interpreter env, CallStack callstack, String default_dir){
049        PlexReader.setDefaultDir(default_dir);
050            }
051    
052            /**
053         Implement setLogging(boolean default_on_flag) command.
054            */
055            public static void invoke(Interpreter env, CallStack callstack, 
056                                boolean default_flag) {
057        PlexReader.setDefaultFlag(default_flag);
058      }
059        
060    }
