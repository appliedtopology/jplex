030    /**
031     * A <code>RipsStream</code> instance is SimplexStream whose elements are
032     * the simplices of the Rips complex of a PointData instance. The
033     * implementation of this class is quite similar to that of
034     * LazyWitnessStream.
035     *
036     * @see        edu.stanford.math.plex.LazyWitnessStream
037     * @see        edu.stanford.math.plex.WitnessStream
038     *
039     * @version $Id: RipsStream.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
040     */
041    public final class RipsStream extends SimplexStream {

043      /**
044       * The PointData underlying this Rips complex.
045       *
046       */
047      public final PointData pdata;
048    
049      /**
050       * The granularity used in assigning filtration indices to simplices.
051       *
052       * <p> A simplex in a Rips complex is regarded as coming into existence
053       * when the "last" of its edges does some. Specifically, we assign a
054       * <i> persistence parameter</i> to each Simplex which is the the
055       * maximum of the persistence parameters assigned to each of the
056       * component edges. The most obvious way to do that is to use the
057       * length of an admissible edge as the persistence parameter of that
058       * edge. (Actually, since the persistence parameters must be integers,
059       * the appropriate index into a sorted array of edge lengths is
060       * used. And recall that we allow there to be a maximum length of the
061       * edges that we consider.) Since this can result in an set of
062       * persistence indices that grows quadratically in the number of
063       * points, we allow the user to control this growth by specifying a
064       * positive granularity. In this case the filtration index of an edge
065       * is the floor of the ratio of the edge length divided by the
066       * granularity. This means that we know that the filtration indices
067       * will never exceed the floor of the ratio of the max edge length
068       * divided by the granularity.
069       *
070       */
071      public final double granularity;
072    
073      /**
074       * The maximum dimension of a Simplex that will be admitted to the complex.
075       */
076      public final int maxDimension;
077    
078      /**
079       * The maximum length of an edge that will be admitted to the
080       * complex. Currently private.
081       */
082      public final double edge_length_limit;

110      /**
111       * Constructor for RipsStream.
112       *
113       * <p>
114       * Make a stream of the simplices in a Rips complex for data.
115       * <p>
116       * @param      double delta Granularity to apply to the underlying metric.
117       * @param      double max   Maximum edge we will allow.
118       * @param      int max_d   Maximum dimension of Simplex instances we construct.
119       * @param      PointData data   Metric data to use in constructing stream.
120       *
121       */
122      RipsStream(double delta, double max, int max_d, PointData data) {
123        assert delta >= 0.0;
124        granularity = delta;
125        edge_length_limit = max;
126        pdata = data;
127        maxDimension = max_d;
128        stack = find_simplices(max, max_d, data);
129      }

154       * Convert a filtration index into a persistence parameter (i.e., double)
155       * -- gets overloaded by some subclasses.
156       *
157       * <p>
158       * @param      fi Filtration index to convert.
159       * @return     double persistence parameter.
160       *
161       */
162      public double convert_filtration_index(int fi) {
163        if (granularity > 0.0) 
164          return granularity * fi;
165        else 
166          return lengths_array[fi];
167      }

195      /**
196       * Return simplices of specified dimension that compare "favorably" with
197       * a given findex.  <p>
198       *
199       * @param      dimension dimenion of interest.
200       * @param      test_index findex we are comparing against.
201       * @param      ct Type of comparison to use.
202       * @return  Simplex[] of matching simplices.
203       */
204      public Simplex[] matchingSimplices(int dimension, int test_index, 
205                                         ComparisonType ct) {
206        Iterator<Simplex> iter = iterator(dimension);
207        ArrayList<Simplex> results = new ArrayList<Simplex>();
208        while(iter.hasNext()) {
209          Simplex n = iter.next();
210          boolean push_it = false;
211          switch(ct) {
212          case LT:
213            push_it = (n.findex() < test_index);
214            break;
215          case LE:
216            push_it = (n.findex() <= test_index);
217            break;
218          case EQ:
219            push_it = (n.findex() == test_index);
220            break;
221          case GE:
222            push_it = (n.findex() >= test_index);
223            break;
224          case GT:
225            push_it = (n.findex() > test_index);
226            break;
227          }
228          if (push_it)
229            results.add(n);
230        }
231        // extract results, put into Simplex[]
232        {
233          Simplex[] return_value = 
234            new Simplex[results.size()];
235          int counter = 0;
236          for(Simplex s : results) 
237            return_value[counter++] = s;
238          // I contend we don't need to sort -- Arrays.sort(return_value);
239          return return_value;
240        }
241      }

243      /**
244       * Return simplices of specified dimension that compare "favorably" with
245       * a given length.  <p>
246       *
247       * @param      dimension dimenion of interest.
248       * @param      test_len length we are comparing against.
249       * @param      ct Type of comparison to use.
250       * @return  Simplex[] of matching simplices.
251       */
252      public Simplex[] matchingSimplices(int dimension, double test_len, 
253                                         ComparisonType ct) {
254        int test_index = 0;
255        if (lengths_array == null)
256          // if we have no lengths_array, just make it into an findex in the
257          // usual way
258          test_index = (int)(Math.floor(test_len/granularity));
259        else {
260          // translate into findex via the lengths_array
261          test_index = Arrays.binarySearch(lengths_array, test_len);
262          if (test_index < 0) {
263            test_index = (-test_index) - 1;
264            // not present -- test_index is now the insertion point for test_len,
265            // so we have to examine the ct argument
266            switch(ct) {
267            case LT:
268            case GE:
269              break;
270            case LE:
271              ct = ComparisonType.LT;
272              break;
273            case GT:
274              ct = ComparisonType.GE;
275              break;
276            case EQ:
277              return new Simplex[0];
278            }
279          }
280        }
281        return matchingSimplices(dimension, test_index, ct);
282      }

297      // Get from (i,j) to the index of the edge_info for the edge from i->j.
298      final static int edge_index(int i, int j, int n) {
299        assert((i > 0) && (i < j) && (j <= n));
300        return (((n * (n - 1)) - ((n - i + 1) * (n - i))) / 2) + (j - (i + 1));
301      }

303      // Test function.
304      static boolean check_edge_index(int n) {
305        int index_limit = (n * (n-1))/2;
306        int prev_index = -1;
307        // System.out.printf("\n\n");
308        for (int i = 1; i <= n; i++) {
309          for (int j = i+1; j <= n; j++) {
310            int current_index = edge_index(i, j, n);
311            // System.out.printf(" %d = edge_index(%d, %d, %d)\n", 
312            // current_index, i, j, n);
313            assert(current_index == (1 + prev_index));
314            assert(current_index < index_limit);
315            assert(current_index >= 0);
316            prev_index = current_index;
317          }
318        }
319        assert(index_limit == (1 + prev_index));
320        //System.out.printf(" >>>>>> LIMIT = %d, last index = %d\n", 
321        // index_limit, prev_index);
322        return true;
323      }

325      final static int all_edges_present(int[] verts, 
326                                         int[] edge_info, int n) {
327        int last_vert = verts[verts.length - 1];
328        int max_findex = 0;
329        for (int i = 0; i < verts.length - 1; i++) {
330          int f_ij = edge_info[edge_index(verts[i], last_vert, n)];
331          if (f_ij < 0)
332            return -1;
333          else if (f_ij > max_findex)
334            max_findex = f_ij;
335        }
336        return max_findex;
337      }

341      // LazyWitnessStream.

344      // The algorithm here is quite simple. We build a SimplexStream.stack
345      // and push all of the Points into it. We then make a queue for storing
346      // Simplices as they are built, and note the beginning (via something
347      // we call a Head). We then we make the edges that are short
348      // enough, adding those to the queue. While adding the edges, we record
349      // the distance information in an array (which we think of as the upper
350      // half of a distance matrix). This "matrix" stores the findex for the
351      // edge from i->j (i < j) or -1 if there is no edge. We then note the
352      // "end of the 1-dimensional simplices" with another Head. Then,
353      // at any point we have Head instances that mark the beginning and
354      // end of the simplices of dimension k. We then iterate over each
355      // Simplex s of dimension k, and for each p > max(s.vertices()), we
356      // check to see if all the edges from p to each vertex exist. If they
357      // do, then we create a new Simplex of dimension k+1 (and if k+1 isn't
358      // the max dimension, we enqueue the new Simplex). This procedure means
359      // that we examine only the possible combinations of vertices in each
360      // dimension of interest, and we only examine a combination once
361      // (because we look at sorted lists).
362      private final SimplexStream.Stack find_simplices(double max_dist, 
363                                                       int max_dimension, 
364                                                       PointData data) {
365        if (max_dimension < 0)
366          return null;
367    
368        // Recall that "count" means that the points are indexed from 1 to N, inclusive.
369        int N = data.count();
370        int[] edge_info = new int[(N * (N - 1)) / 2];
371        int max_findex;
372        if (granularity > 0.0) {
373          max_findex = (int)(Math.floor(max_dist/granularity));
374        } else {
375          max_findex = ((N * (N - 1)) / 2);
376        }
377        // Construct the tail of the queue -- we add any simplices of less
378        // than max dimension to this queue, so that we can go over go over
379        // them again in a later pass
380        SimplexStream.Tail q = new SimplexStream.Tail();
381        // Final repository for the elements of the Rips complex.
382        SimplexStream.Stack stack = new SimplexStream.Stack(max_findex, max_dimension);
383    
384        // Fill in the edge_info array with -1's, which indicate "no edge".
385        for (int i = 0; i < edge_info.length; i++)
386          edge_info[i] = -1;
387    
388        // Add all of the points to the stack.
389        for (int i = 1; i <= N; i++)
390          stack.push(Simplex.makePoint(i, 0));
391       
392        // If max_dimension is 0, we just want the points.
393        if (max_dimension < 1)
394          return stack;
395    
396        // Remember the start of dimension k, where in this case, k=1.
397        SimplexStream.Head last_dim_start = new SimplexStream.Head(q);
398    
399        // Process the edges. We both enqueue them, and create the
400        // edge_info[] which we use to bootstrap from dimension k to
401        // dimension k+1.
402        if (granularity == 0.0) {
403          double[] edge_lens = new double[(N * (N - 1)) / 2];
404          int lengths_counter = 1;
405    
406          // for granularity == 0 we sort the lengths and use the index into
407          // the sorted array of lengths as the filtration index
408          for (int i = 1; i <= N; i++) {
409            for (int j = i+1; j <= N; j++) {
410              double dij = data.distance(i, j);
411              if (dij <= max_dist) {
412                edge_lens[edge_index(i, j, N)] = dij;
413                lengths_counter++;
414              } else 
415                edge_lens[edge_index(i, j, N)] = HUGE;
416            }
417          }
418    
419          // Copy the non-HUGE lengths into the lengths_array (leaving a 0 in
420          // the first entry), and sort it -- we will use the index of dij in
421          // lengths_array as the persistence index for the corresponding edge
422          // simplex. We don't bother looking for duplicate lengths, because
423          // these will occur with probability 0 in real data.
424          lengths_array = new double[lengths_counter];
425          lengths_counter = 1;
426          for (int i = 0; i < edge_lens.length; i++) {
427            if (edge_lens[i] != HUGE)
428              lengths_array[lengths_counter++] = edge_lens[i];
429          }
430          Arrays.sort(lengths_array);
431    
432          // Use binary search to find the index to which the dij value
433          // was moved during sorting. If the performance of this
434          // matters, it would be possible to write a version of the sort
435          // routine that keeps a map of the permutation performed by the
436          // sort, but that seems unlikely to matter here.
437          for (int i = 1; i <= N; i++) {
438            for (int j = i+1; j <= N; j++) {
439              int index = edge_index(i, j, N);
440              double dij = edge_lens[index];
441              if (dij < HUGE) {
442                int pindex = Arrays.binarySearch(lengths_array, dij);
443                assert lengths_array[pindex] == dij;
444                edge_info[index] = pindex;
445                Simplex s = Simplex.makeEdge(i, j, pindex);
446                stack.push(s);
447                q = q.enqueue(s);
448              }
449            }
450          }
451        } else {
452          for (int i = 1; i <= N; i++) {
453            for (int j = i+1; j <= N; j++) {
454              double dij = data.distance(i, j);
455              if (dij <= max_dist) {
456                int f_ij = (int)(Math.floor(dij/granularity));
457                edge_info[edge_index(i, j, N)] = f_ij;
458                Simplex s = Simplex.makeEdge(i, j, f_ij);
459                stack.push(s);
460                q = q.enqueue(s);
461              }
462            }
463          }
464        }
465    
466        // We are ready to construct the higher dimensional data. Since we
467        // already have the edges, we begin with 2-dimensional simplexes,
468        // and continue until we have constructed those of max_dimension.
469        int current_dimension = 2;
470        while (current_dimension <= max_dimension) {
471          // we start processing at last_dim_start
472          SimplexStream.Head current = last_dim_start;
473          // At this point all simplices of current_dimension-1 have been
474          // added to the queue (and none of current_dimension), so we
475          // stop processing for this dimension when current reaches end.
476          SimplexStream.Head end = new SimplexStream.Head(q);
477          // Remember end as last_dim_start, for the next iteration
478          last_dim_start = end;
479          // We construct a scratch array in which to store any potential
480          // vertex sets for possible simplices of dimension
481          // current_dimension.
482          int[] verts = new int[current_dimension + 1];
483    
484          while (current.lessThan(end)) {
485            // Until current reaches end, extract simplices and copy
486            // their vertices into the scratch array.
487            Simplex face = current.nextEntry();
488            assert(face.dimension() == (current_dimension-1));
489            face.vertices(verts);
490            // We know that any k dimensional simplex that we can admit
491            // must have all of its faces in the list of k-1 simplices
492            // (a simplex cannot belong to a complex unless all of its
493            // faces belong, too). That means that any possible k
494            // simplex [x1,...,x(k+1)] must have the face [x1,...,xk]
495            // in the list. So we see that we can get to any possible k
496            // simplex by adding a vertex "onto the end" (that is, the
497            // new vertex is larger than any of the existing ones) of a
498            // k-1 simplex. Further, it is easy to see that adding onto
499            // the end of 2 distinct k-1 simplices can never yield the
500            // same k simplex. So if we just add onto the end of the
501            // simplices in the previous dimension, we will see all
502            // potential k simplices precisely once. So if verts is
503            // [v1,...,vk, <junk>], then we are interested in any
504            // possible simplex [v1,...,vk,x] with x > vk. In this case
505            // we merely need to check if (v1, x),...,(v1, k) are
506            // edges. If so, [v1,...,vk,x] is a simplex, and its
507            // "diameter" is the max of diam([v1,...,vk]) and the
508            // lengths of the edges (v1,x),...(vk,x).
509            int largest_v = verts[current_dimension-1];
510            for (int new_last = largest_v + 1; new_last <= N; new_last++) {
511              verts[current_dimension] = new_last;
512              int f_new = all_edges_present(verts, edge_info, N);
513              if (f_new >= 0) {
514                Simplex newSimplex = 
515                  Simplex.getSimplex(verts, Math.max(face.findex(), f_new));
516                stack.push(newSimplex);
517                if (current_dimension < max_dimension)
518                  // if the dimension of newSimplex isn't maximal,
519                  // then we put it in the queue
520                  q = q.enqueue(newSimplex);
521              }
522            }
523          }
524          // Make sure that if we are exiting we have processed all
525          // simplices in q.
526          assert((current_dimension < max_dimension) || 
527                 (last_dim_start.eql(new SimplexStream.Head(q))));
528          current_dimension++;
529        }
530       
531        return stack;
532      }
