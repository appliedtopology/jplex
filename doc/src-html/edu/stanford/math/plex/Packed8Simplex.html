028    /**
029     * <p>The <code>Packed8Simplex</code> class implements the abstract class
030     * <code>Simplex</code> for Simplices of dimension 6-7 by storing the
031     * vertices as 8 packed positive integer values in three long
032     * integers. This limits us to H6 calculations with this representation,
033     * but it seems very unlikely that we can go even this far an alteration of
034     * the basic algorithm.
035     *
036     * @version $Id: Packed8Simplex.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
037     */
038    public final class Packed8Simplex extends Simplex {
039    
040      public final long bits_lo;
041      public final long bits_lo_mid;
042      public final long bits_hi_mid;
043      public final long bits_hi;
044    
045      // This is the largest vertex index that we can use for this kind of
046      // Simplex.
047      private static final int VERTEX_BIT_SZ = 32;
048      private static final int MAX_8_INDEX = Integer.MAX_VALUE;
049      private static final long DIM_6_MASK = ~((long) MAX_8_INDEX);
050    
051      static boolean checkConstants() {
052        assert(MAX_8_INDEX == 0x7fffffff);
053        assert(DIM_6_MASK == 0xffffffff80000000L);
054        return true;
055      }
056    
057      private static final boolean check_8_vertices(int[] v) {
058        if (v.length <= 6)
059          return false;
060        else if ((v.length == 7) && (v[0] > 0) && (v[1] > v[0]) && (v[2] > v[1]) &&
061                 (v[3] > v[2]) && (v[4] > v[3]) && (v[5] > v[4]) && (v[6] > v[5]))
062          return true;
063        else if ((v.length == 8) && (v[0] > 0) && (v[1] > v[0]) && (v[2] > v[1]) &&
064                 (v[3] > v[2]) && (v[4] > v[3]) && (v[5] > v[4]) && (v[6] > v[5]) && 
065                 (v[7] > v[6]))
066          return true;
067        else
068          return false;
069      }
070    
071      private static final void assert_8_vertices(int[] v) {
072        if (!check_8_vertices(v))
073          throw new 
074            IllegalArgumentException
075            ("Packed8Simplex instances must have either 7 or 8 " + 
076             "distinct positive integer vertices.");
077      }
078    
079      // Convert an array of vertices into the "bits_lo" of a
080      // Packed8Simplex. Vertices MUST BE SORTED.
081      private static final long p8_v_to_l_lo(int[] vertices) {
082        return (((long)vertices[1]) << VERTEX_BIT_SZ) | ((long)vertices[0]);
083      }
084    
085      // Convert an array of vertices into the "bits_lo_mid" of a
086      // Packed8Simplex. Vertices MUST BE SORTED.
087      private static final long p8_v_to_l_lo_mid(int[] vertices) {
088        return (((long)vertices[3]) << VERTEX_BIT_SZ) | ((long)vertices[2]);
089      }
090    
091      // Convert an array of vertices into the "bits_hi_mid" of a
092      // Packed8Simplex. Vertices MUST BE SORTED.
093      private static final long p8_v_to_l_hi_mid(int[] vertices) {
094        return (((long)vertices[5]) << VERTEX_BIT_SZ) | ((long)vertices[4]);
095      }
096    
097      // Convert an array of vertices into the "bits_hi" of a
098      // Packed8Simplex. Vertices MUST BE SORTED.
099      private static final long p8_v_to_l_hi(int[] vertices) {
100        if (vertices.length == 7)
101          return (long) vertices[6];
102        else 
103          return (((long)vertices[7]) << VERTEX_BIT_SZ) | ((long)vertices[6]);
104      }
105    
106      // Extract largest vertex.
107      private final int v8() {
108        return (int) ((bits_hi >>> VERTEX_BIT_SZ) & MAX_8_INDEX);
109      }
110    
111      // Extract 5th vertex (second hi bits vertex).
112      private final int v7() {
113        return (int) (bits_hi & MAX_8_INDEX);
114      }
115    
116      // Extract largest vertex.
117      private final int v6() {
118        return (int) ((bits_hi_mid >>> VERTEX_BIT_SZ) & MAX_8_INDEX);
119      }
120    
121      // Extract 5th vertex (second hi bits vertex).
122      private final int v5() {
123        return (int) (bits_hi_mid & MAX_8_INDEX);
124      }
125    
126      // Extract 4th vertex.
127      private final int v4() {
128        return (int) ((bits_lo_mid >>> VERTEX_BIT_SZ) & MAX_8_INDEX);
129      }
130    
131      // Extract 3rd vertex.
132      private final int v3() {
133        return (int) (bits_lo_mid & MAX_8_INDEX);
134      }
135    
136      // Extract 2nd vertex.
137      private final int v2() {
138        return (int) ((bits_lo >>> VERTEX_BIT_SZ) & MAX_8_INDEX);
139      }
140    
141      // Extract smallest vertex.
142      private final int v1() {
143        return (int) (bits_lo & MAX_8_INDEX);
144      }
145    
146      // We only use the constructors for Packed8Simplex internally. This one
147      // is never used externally because none of bits_lo, bits_mid, or bits_hi
148      // is ever 0.
149      private Packed8Simplex() {
150        bits_lo = 0;
151        bits_lo_mid = 0;
152        bits_hi_mid = 0;
153        bits_hi = 0;
154      }
155    
156      // We only use the constructors for Packed8Simplex internally. 
157      private Packed8Simplex(long bits_lo_val, long bits_lo_mid_val,
158                             long bits_hi_mid_val, long bits_hi_val) {
159        bits_lo = bits_lo_val;
160        bits_lo_mid = bits_lo_mid_val;
161        bits_hi_mid = bits_hi_mid_val;
162        bits_hi = bits_hi_val;
163      }
164    
165      // In the explicit constructors, the indices must all be non-zero and
166      // be in increasing order.
167      private Packed8Simplex(int v1, int v2, int v3, int v4, int v5, int v6, 
168                             int v7) {
169        assert((v7 > v6) && (v6 > v5) && (v5 > v4) && (v4 > v3) && 
170               (v3 > v2) && (v2 > v1) && (v1 > 0));
171        bits_hi = (long) v7;
172        bits_hi_mid = (((long) v6) << VERTEX_BIT_SZ) | ((long) v5);
173        bits_lo_mid = (((long) v4) << VERTEX_BIT_SZ) | ((long) v3);
174        bits_lo = (((long) v2) << VERTEX_BIT_SZ) | ((long) v1);
175      }
176      private Packed8Simplex(int v1, int v2, int v3, int v4, int v5, int v6, 
177                             int v7, int v8) {
178        assert((v8 > v7) && (v7 > v6) && (v6 > v5) && (v5 > v4) && 
179               (v4 > v3) && (v3 > v2) && (v2 > v1) && (v1 > 0));
180        bits_hi = (((long) v8) << VERTEX_BIT_SZ) | ((long) v7);
181        bits_hi_mid = (((long) v6) << VERTEX_BIT_SZ) | ((long) v5);
182        bits_lo_mid = (((long) v4) << VERTEX_BIT_SZ) | ((long) v3);
183        bits_lo = (((long) v2) << VERTEX_BIT_SZ) | ((long) v1);
184      }

187      /**
188       * Overrides Object hashcode.
189       *
190       * <p>
191       * @return     CRC hash of the vertex set.
192       *
193       */
194      public final int hashCode() {
195        return CRC.hash32(bits_hi, 
196                          CRC.hash32(bits_hi_mid, 
197                                     CRC.hash32(bits_lo_mid, 
198                                                CRC.hash32(bits_lo, 0))));
199      }
200    
201      /**
202       * Overrides Object equals.
203       *
204       * <p>
205       * @param      obj   object to compare.
206       * @return true or false, depending on whether or not the Simplex is =
207       *                        to obj.
208       *
209       */
210      public final boolean equals(Object obj) {
211        if (!(obj instanceof Packed8Simplex))
212          return false;
213        Packed8Simplex s = (Packed8Simplex) obj;
214        return ((bits_lo == s.bits_lo) && (bits_lo_mid == s.bits_lo_mid) && 
215                 (bits_hi_mid == s.bits_hi_mid) && (bits_hi == s.bits_hi));
216      }
217    
218      // Internal comparison routine.
219      private final int compareTo(Packed8Simplex s) {
220        if (bits_hi > s.bits_hi)
221          return 1;
222        else if (bits_hi < s.bits_hi)
223          return -1;
224        else if (bits_hi_mid > s.bits_hi_mid)
225          return 1;
226        else if (bits_hi_mid < s.bits_hi_mid)
227          return -1;
228        else if (bits_lo_mid > s.bits_lo_mid)
229          return 1;
230        else if (bits_lo_mid < s.bits_lo_mid)
231          return -1;
232        else if (bits_lo > s.bits_lo)
233          return 1;
234        else if (bits_lo < s.bits_lo)
235          return -1;
236        else
237          return 0;
238      }
239    
240      /**
241       * Implements Comparable interface.
242       *
243       * <p>
244       * @param      s   Simplex to compare.
245       * @return     negative, 0, or positive, if this <, =, resp. > than s.
246       *
247       */
248      public final int compareTo(Simplex s) {
249        // The Packed8Simplex instances are of larger dimension than any
250        // other kind of Simplex, so if s is a Simplex but not a
251        // Packed8Simplex, then this > s.
252        if (!(s instanceof Packed8Simplex))
253          return 1;
254        else 
255          return this.compareTo((Packed8Simplex)s);
256      }
257    
258      /**
259       * Simplex Less Than. Used internally in Persistence and Chain
260       * code. Not the same as compareTo.
261       *
262       * <p>
263       * @param      s   Simplex to compare.
264       * @return     true if this < s, else false.
265       *
266       */
267      final boolean slt(Simplex s) {
268        if (this.findex() != s.findex())
269          return (this.findex() < s.findex());
270        else if (s instanceof Packed8Simplex) {
271          Packed8Simplex ps = (Packed8Simplex) s;
272          if (bits_hi < ps.bits_hi)
273            return true;
274          else if (bits_hi == ps.bits_hi) {
275            if (bits_hi_mid < ps.bits_hi_mid)
276              return true;
277            else if (bits_hi_mid == ps.bits_hi_mid) {
278              if (bits_lo_mid < ps.bits_lo_mid)
279                return true;
280              else if (bits_lo_mid == ps.bits_lo_mid)
281                return (bits_lo < ps.bits_lo);
282              else
283                return false;
284            } else 
285              return false;
286          } else
287            return false;
288        } else 
289          return (this.dimension() < s.dimension());
290      }
291    
292      /**
293       * Simplex EQuals. Used internally in Persistence and Chain code.  
294       *
295       * <p>
296       * @param      s   Simplex to compare.
297       * @return     true is equal, else false.
298       *
299       */
300      final boolean seq(Simplex s) {
301        if (!(s instanceof Packed8Simplex))
302          return false;
303        Packed8Simplex ps = (Packed8Simplex) s;
304        return ((bits_lo == ps.bits_lo) && (bits_lo_mid == ps.bits_lo_mid) && 
305                (bits_hi_mid == ps.bits_hi_mid) && (bits_hi == ps.bits_hi));
306      }

308      /**
309       * Make a "blank" copy of the Simplex -- equivalent to getSimplex(this.vertices()).
310       * <p>
311       *
312       *
313       * @return     the copied instance.
314       *
315       */    
316      public final Simplex copy() {
317        return new Packed8Simplex(bits_lo, bits_lo_mid, bits_hi_mid, bits_hi);
318      }

320      /**
321       * Returns the dimension of self.
322       * <p>
323       *
324       *
325       * @return     int, the dimension of the simplex.
326       *
327       * @see        edu.stanford.math.plex.Simplex#dimension
328       */    
329      public final int dimension() { 
330        if ((bits_hi & DIM_6_MASK) == 0) 
331          return 6;
332        else
333          return 7;
334      }
335    
336      /**
337       * Returns the indices of self as an array.
338       * <p>
339       *
340       *
341       * @return     an int[] of vertices, or null if no vertices.
342       *
343       * @see        edu.stanford.math.plex.Simplex#vertices
344       */    
345      public final int[] vertices() { 
346        final int dimension = dimension();
347        int verts[] = new int[dimension+1];
348    
349        verts[0] = this.v1();
350        verts[1] = this.v2();
351        verts[2] = this.v3();
352        verts[3] = this.v4();
353        verts[4] = this.v5();
354        verts[5] = this.v6();
355    
356    
357        if (dimension == 6) {
358          verts[6] = this.v7();
359          return verts;
360        }
361    
362        verts[6] = this.v7();
363        verts[7] = this.v8();
364        return verts;
365      }
366    
367    
368      /**
369       * Returns the indices of self in the given array argument.
370       * <p>
371       *
372       *
373       * @param      verts   the int[] into which the vertices, if any, are written.
374       * @return     the given array argument, or null if no vertices.
375       *
376       * @see        edu.stanford.math.plex.Simplex#vertices
377       */
378      public final int[] vertices(int[] verts) { 
379        final int dimension = dimension();
380    
381        verts[0] = this.v1();
382        verts[1] = this.v2();
383        verts[2] = this.v3();
384        verts[3] = this.v4();
385        verts[4] = this.v5();
386        verts[5] = this.v6();
387    
388        if (dimension == 6) {
389          verts[6] = this.v7();
390          return verts;
391        }
392    
393        verts[6] = this.v7();
394        verts[7] = this.v8();
395        return verts;
396      }

398      // The obvious thing.
399      public static final Simplex makeSimplex(int[] vertices) {
400        Simplex.vertex_sort(vertices);
401        assert_8_vertices(vertices);
402        return new Packed8Simplex(p8_v_to_l_lo(vertices), 
403                                  p8_v_to_l_lo_mid(vertices), 
404                                  p8_v_to_l_hi_mid(vertices), 
405                                  p8_v_to_l_hi(vertices));
406      }
407    
408      // The other obvious thing.
409      public static final Simplex makeSimplexPresorted(int[] vertices) {
410        assert_8_vertices(vertices);
411        return new Packed8Simplex(p8_v_to_l_lo(vertices), 
412                                  p8_v_to_l_lo_mid(vertices), 
413                                  p8_v_to_l_hi_mid(vertices), 
414                                  p8_v_to_l_hi(vertices));
415      }

417      /**
418       * Returns the boundary of self.  
419       *
420       * <p> It is simple enough to do explicitly, and it needs to be fast.
421       *
422       *
423       * @return     [face0, face1, ...]
424       *
425       * @see        edu.stanford.math.plex.Simplex#vertices
426       */
427      public final Simplex[] boundary() {
428        final int dimension = dimension();
429        Simplex[] return_value = new Simplex[dimension+1];
430    
431        if (dimension == 6) {
432          return_value[0] = new Packed6Simplex(this.v2(), this.v3(), this.v4(), 
433                                               this.v5(), this.v6(), this.v7());
434          return_value[1] = new Packed6Simplex(this.v1(), this.v3(), this.v4(), 
435                                               this.v5(), this.v6(), this.v7());
436          return_value[2] = new Packed6Simplex(this.v1(), this.v2(), this.v4(), 
437                                               this.v5(), this.v6(), this.v7());
438          return_value[3] = new Packed6Simplex(this.v1(), this.v2(), this.v3(), 
439                                               this.v5(), this.v6(), this.v7());
440          return_value[4] = new Packed6Simplex(this.v1(), this.v2(), this.v3(), 
441                                               this.v4(), this.v6(), this.v7());
442          return_value[5] = new Packed6Simplex(this.v1(), this.v2(), this.v3(), 
443                                               this.v4(), this.v5(), this.v7());
444          return_value[6] = new Packed6Simplex(this.v1(), this.v2(), this.v3(), 
445                                               this.v4(), this.v5(), this.v6());
446        } else {
447          return_value[0] = 
448            new Packed8Simplex(this.v2(), this.v3(), this.v4(), 
449                               this.v5(), this.v6(), this.v7(), this.v8());
450          return_value[1] = 
451            new Packed8Simplex(this.v1(), this.v3(), this.v4(), 
452                               this.v5(), this.v6(), this.v7(), this.v8());
453          return_value[2] = 
454            new Packed8Simplex(this.v1(), this.v2(), this.v4(), 
455                               this.v5(), this.v6(), this.v7(), this.v8());
456          return_value[3] = 
457            new Packed8Simplex(this.v1(), this.v2(), this.v3(), 
458                               this.v5(), this.v6(), this.v7(), this.v8());
459          return_value[4] = 
460            new Packed8Simplex(this.v1(), this.v2(), this.v3(), 
461                               this.v4(), this.v6(), this.v7(), this.v8());
462          return_value[5] = 
463            new Packed8Simplex(this.v1(), this.v2(), this.v3(), 
464                               this.v4(), this.v5(), this.v7(), this.v8());
465          return_value[6] = 
466            new Packed8Simplex(this.v1(), this.v2(), this.v3(), 
467                               this.v4(), this.v5(), this.v6(), this.v8());
468          return_value[7] = 
469            new Packed8Simplex(this.v1(), this.v2(), this.v3(), 
470                               this.v4(), this.v5(), this.v6(), this.v7());
471        }
472        return return_value;
473      }
