029    /**
030     * A <code>WitnessStream</code> is an iterable Witness complex.
031     *
032     * <p>A WitnessStream is a SimplexStream whose elements are the simplices
033     * of the Witness complex of a PointData instance.  That is, a simplex is
034     * added to the complex only each face is present and a witness is found
035     * for the entire simplex.  The details of this construction are described
036     * in <i>Topological estimation using witness complexes</i>, Vin de Silva
037     * and Gunnar Carlsson, Symposium on Point-Based Graphics, ETH, Zurich,
038     * Switzerland, June 2-4, 2004. 
039     *
040     * @see        edu.stanford.math.plex.LazyWitnessStream
041     *
042     * @version $Id: WitnessStream.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
043     */
044    /**
045     * The <code>WitnessStream</code> class implements Witness complex
046     * creation.
047     *
048     * @version $Id: WitnessStream.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
049     */
050    public final class WitnessStream extends SimplexStream {

052      /**
053       * Randomly choose a subset of a given PointData for use as landmarks.
054       *
055       * <p>
056       * @param      data PointData from which to choose landmarks.
057       * @param      L number of landmarks to choose.
058       * @return     An int[] containing indices of the landmarks.
059       *
060       */
061      public static int[] makeRandomLandmarks(PointData data, int L) {
062        assert(L <= data.count());
063        int N = data.count();
064        // landmarks[0] is always 0
065        int[] landmarks = new int[L+1];
066        HashSet<Integer> used = new HashSet<Integer>(L);
067        Random rand = new Random(0);
068        for(int counter = 1; counter < landmarks.length; ){
069          int possible = rand.nextInt(N) + 1;
070          Integer pInteger = new Integer(possible);
071          if (!used.contains(pInteger)) {
072            used.add(pInteger);
073            landmarks[counter++] = possible;
074          }
075        }
076        return landmarks;
077      }

079      /**
080       * Estimate Rmax for a given PointData and landmark set.
081       *
082       * <p>
083       * @param      data The entire data set.
084       * @param      landmarks The landmark set.
085       * @return     The max/min of the landmarks to points distances.
086       *
087       */
088      public static double estimateRmax(PointData data, int[] landmarks) {
089        int L = landmarks.length - 1;
090        int N = data.count();
091        double Rmax = 0.0;
092    
093        for (int n = 1; n <= N; n++) {
094          // for each point, find the closest landmark
095          double m_n = HUGE;
096          for (int l = 1; l <= L; l++) {
097            m_n = min(m_n, distance_ln(l, n, landmarks, data));
098          }
099          // then take the max over all points of the closest landmark
100          Rmax = max(m_n, Rmax);
101        }
102        return Rmax;
103      }

137      /**
138       * Constructor for WitnessStream.
139       *
140       * <p>
141       * Make a stream of the simplices in a Witness complex for data.
142       * <p>
143       * @param      delta Granularity to apply to the underlying metric.
144       * @param      max_d   Maximum dimension of Simplex instances we construct.
145       * @param      R   Maximum R we will use.
146       * @param      landmarks Indices of points to use as landmarks. The
147       *                        simplices constructed will use the landmark
148       *                        indices, rather than the indices into data.
149       * @param      PointData data   Metric data to use in constructing stream.
150       *
151       */
152      WitnessStream(double delta, int max_d, double R, 
153                    int[] landmarks, PointData data) {
154        assert delta > 0.0;
155        granularity = delta;
156        this.landmarks = landmarks;
157        maxR = R;
158        pdata = data;
159        maxDimension = max_d;
160        stack = find_simplices(delta, R, max_d, landmarks, data);
161      }

219      // Implementation note: In the code below, L is the number of landmarks
220      // and N is the number of data points. To make the code match the
221      // papers a bit more, we have the indices go from 1 to L (or N),
222      // instead of from 0 to L-1 (or N-1), which would be the normal way in
223      // C (and hence C++ and Java). So the 0-th entry of things in nothing,
224      // and when we make arrays that get indexed directly, they are 1
225      // longer. When we index matrices we create a 1-array and then have an
226      // indexing function that adjusts the arguments (that is, we can make
227      // the 1-array exactly the right length). Since Java should inline
228      // these indexing functions, and the compiler can't deal with 2-arrays
229      // of fixed size (as it would in C or C++), I chose to implement
230      // matrices this way.
231      //
232      // When there is at most one of each, I use l as the index over
233      // landmarks and n as the index over data. (I have found it hard to
234      // tell l from 1 in the code a couple of times while writing it, so
235      // don't feel bad if you make the same mistake reading it.)
236    
237    
238      // Get from (n,k) to the index of m(n, k). See the paper referenced
239      // above for meaning of m[]. The k index is an ordinal, not any of the
240      // l/n/d values that we've mentioned up to now.
241      private final static int m_index(int n, int k, int N) {
242        assert((k > 0) && (1 <= n) && (n <= N));
243        return ((k-1) * N) + (n-1);
244      }

251      // Compute distance matrix entry for given landmark/data_point.
252      static final double distance_ln(int l, int n, int[] landmarks, PointData data) {
253        if (landmarks[l] == n)
254          // exclude the landmark from consideration
255          return HUGE;
256        else
257          return data.distance(landmarks[l], n);
258      }
259    
260      static final double HUGE = Double.MAX_VALUE/2; 
261    
262      // D[] index for the distance from landmark l to data point n.
263      final static int D_index(int l, int n, int L, int N) {
264        assert((l > 0) && (l <= L) && (n > 0) && (n <= N));
265        return ((l-1) * N) + (n-1);
266      }

270        if (x < y)
271          return x;
272        else
273          return y;
274      }

276      final static double max(double x, double y) {
277        if (x > y)
278          return x;
279        else
280          return y;
281      }

283      final static int max(int x, int y) {
284        if (x > y)
285          return x;
286        else
287          return y;

296      // The algorithm here is a bit different than the the one for Rips
297      // complexes, but it is still an iterative process in increasing
298      // dimension of the simplices being constructed. A (non-lazy) Witness
299      // complex contains a simplex s iff the complex contains each of the
300      // faces of s, and there is a witness for s itself. The idea is to
301      // start off adding the landmarks, and building the distance matrix, as
302      // we do in LazyWitnessStream. However, after that we construct an
303      // array m[][] of double[max_dimension] of the 1st, 2nd,
304      // ... max_dimension-th smallest distances for each of the 1<=n<=N pts
305      // in data.  Then, we construct the edge set just as for
306      // LazyWitnessStream with nu=2. Now, at each step for dim = d, we have
307      // the d-1 dimensional simplices, each with there filtration indices
308      // set, stored in the queue. We then create an intering table for this
309      // set of d-1 dimensional "faces", and iterate over the d-dimensional
310      // simplices just as in RipsStream or LazyWitnessStream. However, the
311      // first step in processing a d-dimensional simplex is to compute its
312      // boundary, and confirm that each face is in the interning table. If
313      // this succeeds we will return the maximum filtration index of all of
314      // the faces. If it fails, we go onto the next d-dimensional simplex,
315      // but if it succeeds, we must find the 1<=n<=N with the smallest
316      // max(dist(i, p1), ... dist(i, pd)). We take this value, subtract from
317      // it the entry in m[k-1][i] (the k-th smallest distance, indexed from
318      // 0 to max_dimension-1). If this difference Rs is <= R_max, we accept
319      // the simplex, and the filtration index is the maximum of the maximum
320      // of the faces and floor(max(Rs, 0.0)/delta).  Note that this
321      // iteration is more expensive both because it is necessary to
322      // construct m[][], but also because the last step of acceptance
323      // requires O(k*N) operations. I see no way of avoiding this.
324      private final SimplexStream.Stack 
325        find_simplices(double delta, double R_max, int max_dimension, 
326                       int[] landmarks, PointData data) {
327    
328        // Recall that we never use 0 as a simplex index, so landmarks[0]
329        // should be empty
330        assert(landmarks[0] == 0);
331        // there is no point doing this unless 1 < L <= N and max_dimension <= L.
332        assert((1 < (landmarks.length-1)) && 
333               (max_dimension <= (landmarks.length-1)) && 
334               ((landmarks.length-1) < data.count()));
335    
336        if (max_dimension < 0)
337          return null;
338    
339        // The landmarks are indexed from 1 to L, inclusive, and since 0 is excluded,
340        // L = landmarks.length-1.
341        int L = landmarks.length - 1;
342        int max_findex = (int)(Math.floor(R_max/granularity));
343        // Construct the tail of the queue -- we add any simplices of
344        // dimension d < max_dimension to this queue, so that we can
345        // iterate over them them in constructing the simplices of
346        // dimension d+1.
347        SimplexStream.Tail q = new SimplexStream.Tail();
348        // Final repository for the elements of the Witness complex.
349        SimplexStream.Stack stack = 
350          new SimplexStream.Stack(max_findex, max_dimension);
351        // Recall that "count" means that the points are indexed from 1 to
352        // N, inclusive.
353        int N = data.count();
354        // we can think of this as 2-dimensional, but it's (supposed to me)
355        // simpler and faster to convert the two indices to a single one
356        // via an index function, so that's what we'll do
357        double[] m = new double[N * (max_dimension+1)];
358        double[] D = new double[L * N];
359    
360        // fill in the distance matrix
361        for (int l = 1; l <= L; l++)
362          for (int n = 1; n <= N; n++)
363            D[D_index(l, n, L, N)] = distance_ln(l, n, landmarks, data);
364             
365    
366        // Add all of the landmark Points (0-simplices) to the stack.
367        for (int l = 1; l <= L; l++)
368          stack.push(Simplex.makePoint(l, 0));
369       
370        // If max_dimension is 0, we just want the Points.
371        if (max_dimension < 1)
372          return stack;
373    
374        // Remember the start of dimension k, where in this case, k=1.
375        SimplexStream.Head last_dim_start = new SimplexStream.Head(q);
376        // Remember how many d-1 dimensional simplices are being
377        // made. These are the faces of the d dimensional simplices, and we
378        // want to intern all possible faces to make checking faster. See
379        // below for more details.
380        int faces_to_intern = 0;
381    
382        // construct the metric info m[] and process the edges
383        {
384          // Now we make m[] -- we want the smallest max_dimension
385          // entries from each column of D.
386          {
387            double[] m_tmp = new double[L+1];
388    
389            for (int n = 1; n <= N; n++) {
390              for (int l = 1; l <= L; l++) 
391                m_tmp[l] = D[D_index(l, n, L, N)];
392              // sort the initial contents of m_tmp -- it might be
393              // faster to do this by hand, but it is a tiny amount
394              // of the time spent in this function, and when I tried
395              // once to do it by hand, I messed it up
396              Arrays.sort(m_tmp);
397              assert(m_tmp[0] == 0.0);
398    
399              // copy the contents of m_tmp into m
400              for(int d = 1; d <= max_dimension+1; d++) {
401                assert(m_tmp[d] < HUGE/2.0);
402                m[m_index(n, d, N)] = m_tmp[d];
403              }
404            }
405          }
406    
407          // Process the edges. Unlike RipsStream and LazyWitnessStream,
408          // there is no special preprocessing going on at this step. The
409          // only special thing about this case is that we know that ALL
410          // of the faces of the edges (that is, all the vertices) are in
411          // the complex, so we can skip the "intern the k-1 dimensional
412          // simplices and then see if they exist" step here. But other
413          // than that, the process of admitting an edge is the same as
414          // the higher dimensional cases. We keep it here because we
415          // need to start the dimensional iteration someplace, and
416          // because it emphasizes the similarities between building this
417          // type of complex and the lazy version. Here i and j are the
418          // landmark indices -- this seemed better than using l1 and l2.
419          for (int i = 1; i <= L; i++) {
420            for (int j = i+1; j <= L; j++) {
421              // compute the R at which the edge comes into existence
422              // without the m[] adjustment
423              double e_ij = HUGE;
424              assert(e_ij >= 0.0);
425              for (int n = 1; n <= N; n++) {
426                double tmp_ijn = 
427                  max(D[D_index(i, n, L, N)], D[D_index(j, n, L, N)]) - 
428                  m[m_index(n, 2, N)];
429                assert(tmp_ijn >= 0.0);
430                e_ij = min(e_ij, tmp_ijn);
431              }
432              // if the edge appears early enough, create it, and
433              // then push and enqueue it
434              if (e_ij <= R_max) {
435                int f_ij = (int)(Math.floor(e_ij/granularity));
436                Simplex s = Simplex.makeEdge(i, j, f_ij);
437                stack.push(s);
438                q = q.enqueue(s);
439                // count the edges -- these will be all possible
440                // faces for the triangles, and we will be able to
441                // immediately exclude any triangle whose boundary
442                // has an edge not in this set
443                faces_to_intern++;
444              }
445            }
446          }
447        }
448    
449        // We are ready to construct the higher dimensional data. Since we
450        // already have the edges, we begin with 2-dimensional simplexes,
451        // and continue until we have constructed those of max_dimension.
452        int current_dimension = 2;
453        while (current_dimension <= max_dimension) {
454          // we start processing at last_dim_start
455          SimplexStream.Head current = last_dim_start;
456          SimplexStream.Head interning_current = current.copy();
457          // At this point all simplices of current_dimension-1 have been
458          // added to the queue (and none of current_dimension), so we
459          // stop processing for this dimension when current reaches end.
460          SimplexStream.Head end = new SimplexStream.Head(q);
461          // Remember end as last_dim_start, for the next iteration
462          last_dim_start = end;
463          // We construct a scratch array in which to store any potential
464          // vertex sets for possible simplices of dimension
465          // current_dimension.
466          int[] verts = new int[current_dimension + 1];
467          // Make a table of all of the current_dimension-1 faces -- for
468          // any constructed current_dimension simplex, we can quickly
469          // check that all of its faces are present by taking the
470          // boundary and verifying that each face is in the table.
471          SimplexTable faces = new SimplexTable(faces_to_intern);
472          // to avoid repeatedly looking up distances for the faces vertices,
473          // we make an array in which to cache these values
474          double[] face_R = new double[N+1];
475          // indicate whether this has been initialize for this face
476          boolean face_R_initialized = false;
477    
478          // reset the counter
479          faces_to_intern = 0;
480    
481          while (interning_current.lessThan(end)) {
482            // Until interning_current reaches end, extract simplices
483            // and put them in the faces table.
484            Simplex face = interning_current.nextEntry();
485            faces.put(face);
486          }
487    
488          while (current.lessThan(end)) {
489            // Until current reaches end, extract simplices and copy
490            // their vertices into the scratch array.
491            Simplex face = current.nextEntry();
492            assert(face.dimension() == (current_dimension-1));
493            face.vertices(verts);
494    
495            face_R_initialized = false;
496    
497            // We know that any d dimensional simplex that we can admit
498            // must have all of its faces in the list of d-1 simplices
499            // (a simplex cannot belong to a complex unless all of its
500            // faces belong, too). That means that any possible d
501            // simplex [x1,...,x(d+1)] must have the face [x1,...,xd]
502            // in the list. So we see that we can get to any possible d
503            // simplex by adding a vertex "onto the end" (that is, the
504            // new vertex is larger than any of the existing ones) of a
505            // d-1 simplex. Further, it is easy to see that adding onto
506            // the end of 2 distinct d-1 simplices can never yield the
507            // same d simplex. So if we just add onto the end of the
508            // simplices in the previous dimension, we will see all
509            // potential d simplices precisely once. So if verts is
510            // [x1,...,xd, <junk>], then we are interested in any
511            // possible simplex [x1,...,xd,x] with x > xd. In this case
512            // we must first take [x1,...,xd,x].boundary() and see if
513            // all of the entries are in the faces table (although, of
514            // course, we could ignore the first entry). Then if all
515            // faces are present, we have to search for the smallest R
516            // for which a witness exists for the the simplex. If that
517            // R is less than R_max, then the simplex is added, and
518            // it's filtration index is the min of the filtration
519            // indices of the faces and floor(R/granularity).
520            int largest_v = verts[current_dimension-1];
521            for (int new_last = largest_v + 1; new_last <= L; new_last++) {
522              verts[current_dimension] = new_last;
523              Simplex possible = Simplex.getSimplexPresorted(verts);
524              Simplex[] bdy = possible.boundary();
525              int max_face_fi = 0;
526              for(int fi = 0; fi < bdy.length; fi++) {
527                Simplex interned_face = faces.get(bdy[fi]);
528                if (interned_face != null)
529                  max_face_fi = max(max_face_fi, interned_face.findex());
530                else {
531                  max_face_fi = -1;
532                  break;
533                }
534              }
535              if (max_face_fi >= 0) {
536                double possible_R = HUGE;
537                // Look for a witness for "possible" -- this part
538                // of the code is where a large majority of the time is
539                // spent. Is there some better way to do this?
540    
541                if (!face_R_initialized) {
542                  // this hack seems to save about 30% of the time
543                  // initialize face_R for the new face
544                  for (int n = 1; n <= N; n++) {
545                    double R_n = 0.0;
546                    for(int vi = 0; vi < current_dimension; vi++)
547                      R_n = max(R_n, D[D_index(verts[vi], n, L, N)]);
548                    face_R[n] = R_n;
549                  }
550                  face_R_initialized = true;
551                }
552    
553                for (int n = 1; n <= N; n++) {
554                  double R_n =
555                    max(face_R[n], 
556                        D[D_index(verts[current_dimension], n, L, N)]);
557                  R_n -= m[m_index(n, current_dimension+1, N)];
558                  assert(R_n >= 0.0);
559                  possible_R = min(possible_R, R_n);
560                }
561                if (possible_R <= R_max) {
562                  possible.setfindex(max(max_face_fi, 
563                                         (int)Math.floor(possible_R/delta)));
564                  stack.push(possible);
565                  if (current_dimension < max_dimension) {
566                    // if the dimension of possible isn't
567                    // maximal, then we put it in the queue and
568                    // count it for interning the next time
569                    q = q.enqueue(possible);
570                    faces_to_intern++;
571                  }
572                }
573              }
574            }
575          }
576          // Make sure that if we are exiting we have processed all
577          // simplices in q.
578          assert((current_dimension < max_dimension) || 
579                 (last_dim_start.eql(new SimplexStream.Head(q))));
580          current_dimension++;
581        }
582       
583        return stack;
584      }
