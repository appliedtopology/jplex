049      @After

053            
054      @Test
055        public void testSomeBehavior() {
056        int[] vertices = new int[] {1, 3, 4, 2};
057        int[] v6 = new int[] {1, 3, 4, 2, 6, 8};
058        Simplex test = Simplex.getSimplex(vertices);
059        Simplex t6 = Simplex.getSimplex(v6);
060        int[] tverts = test.vertices();
061        int[] t6verts = t6.vertices();
062        assertTrue("s.vertices equals original vertices", 
063                   Plex.equalPtArrays(Simplex.getSimplex(vertices).vertices(),
064                                      Simplex.vertex_sort(vertices)));
065        assertTrue("s.vertices equals original vertices", 
066                   Plex.equalPtArrays(Simplex.getSimplex(v6).vertices(),
067                                      Simplex.vertex_sort(v6)));
068        Simplex[] test_bdy = test.boundary();
069        Simplex[] t6_bdy = t6.boundary();
070        assertTrue("second bdy elt is correct",
071                   test_bdy[1].equals(Simplex.getSimplex(new int[] {1, 3, 4})));
072        assertTrue("second bdy elt is correct",
073                   t6_bdy[1].equals(Simplex.getSimplex(new int[] {1, 3, 4, 6, 8})));
074        Simplex e2 = Simplex.makeEdge(1, Integer.MAX_VALUE, -1);
075        assertTrue("e2 is okay", e2.dimension() == 1);
076        assertTrue("e2 is okay", e2 instanceof Packed2Simplex);
077      }
