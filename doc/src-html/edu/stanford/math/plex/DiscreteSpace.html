026    package edu.stanford.math.plex;
027    
028    /**
029     * The <code>DiscreteSpace</code> class implements a finite discrete metric
030     * space. Used for testing.
031     *
032     * @version $Id: DiscreteSpace.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
033     */
034    public class DiscreteSpace extends PointData {
035    
036      public final int N;
037    
038      /**
039       * The number of data points.
040       *
041       * <p>
042       *
043       * @return     the number of points
044       */
045      public final int count() {
046        return N;
047      }
048       
049      /**
050       * Discrete distance.
051       *
052       * <p>
053       *
054       * @param      p1   the first point
055       * @param      p2   the second point
056       * @return     the distance between p1 and p2.
057       */
058      public final double distance (int p1, int p2) {
059        if (p1 == p2)
060          return 0.0;
061        else 
062          return 1.0;
063      }
064    
065      // Constructor for this class.
066      public DiscreteSpace(int N) {
067        this.N = N;
068      }
069    
070    }
