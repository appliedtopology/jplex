029    /**
030     * An instance of the abstract class <code>SimplexStream</code> provides
031     * the simplices of a filtered complex in lexicographic order on
032     * persistence index and dimension. The ordering is unspecified beyond that
033     * point, and may even vary from instance to instance. The reason that
034     * SimplexStream has both <code>hasNext()</code> and <code>next()</code>
035     * methods <em>and</em> implements the <code>Iterable</code> interface is
036     * that the internal iteration methods are allowed to be destructive. That
037     * is, it is possible (and in fact does happen) that
038     * <code>stream.next()</code> will remove simplices from
039     * <code>stream</code>, making it impossible to use this method to iterate
040     * over <code>stream</code> a second time. Therefore, unless you are
041     * prepared to <em>consume</em> the stream, use the <code>Iterable</code>
042     * interface.
043     *
044     * @version $Id: SimplexStream.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
045     */
046    
047    public abstract class SimplexStream implements Iterable<Simplex> {
048

050       * Is there a next Simplex in the stream?

054       *

059       * Next Simplex in the stream.
060       *
061       * <p>
062       * @return     Return the smallest remaining Simplex instance.
063       *
064       */
065      public abstract Simplex next();

068      /**
069       * How many simplices are in the stream?
070       *
071       * <p>
072       * @return     The number of simplices in the stream.
073       *
074       */
075      public abstract int size();

077      /**
078       * Max dimension of simplices in the stream.
079       *
080       * <p>
081       * @return     Max dimension of simplices in the stream.

084      public abstract int maxDimension();

086      /**
087       * Convert a PersistenceInterval into Float format in a stream-specific manner.
088       *
089       * <p>
090       * @param      p PersistenceInterval to convert.
091       * @return     Converted PersistenceInterval.Float.
092       *
093       */
094      public PersistenceInterval.Float convertInterval(PersistenceInterval p) {
095        if (p instanceof PersistenceInterval.Float)
096          return (PersistenceInterval.Float) p;
097        else {
098          PersistenceInterval.Int pi = (PersistenceInterval.Int) p;
099          double start = convert_filtration_index(pi.start);
100          double end = p.infiniteExtent()?
101            PersistenceInterval.Float.MAX_END:
102            convert_filtration_index(pi.end);
103          return new PersistenceInterval.Float(pi.dimension, start, end);
104        } 
105      }
106    
107      /**
108       * Convert a filtration index into a persistence parameter (i.e., double)
109       * -- gets overloaded by some subclasses.
110       *
111       * <p>
112       * @param      fi Filtration index to convert.
113       * @return     double persistence parameter.
114       *
115       */
116      public double convert_filtration_index(int fi) {
117        return (double) fi;
118      }

121      /**
122       * Check the stream as much as possible.
123       *
124       * <p>
125       * @return     True if the stream is consistent, else false.
126       *
127       */
128      public boolean verify() {
129        SimplexTable faces = new SimplexTable(size());
130        for(Simplex s : this) {
131          Simplex[] bdy = s.boundary();
132          if (bdy != null) {
133            for(Simplex f : bdy) {
134              Simplex tmp = faces.get(f);
135              if ((tmp == null) || (tmp.findex() > s.findex()))
136                return false;
137            }
138          }
139          faces.put(s);
140        }
141        return true;
142      }

144      public enum ComparisonType { LT, LE, EQ, GE, GT };

146      // As we are constructing the stream contents, we need to temporarily
147      // store simplices in a FIFO queue. Since we want to deal with the case
148      // where we have a LOT of simplices in the stream, we make some
149      // special-purpose classes with a very small amounts of overhead. NOTE:
150      // The FIFO queue functionality requires using both the Tail, to
151      // which objects are added, and the Head, from which they are
152      // extracted.
153    
154      /**
155       * An instance of <code>Tail</code> provides the FI part of a FIFO
156       * queue. To retrieve the enqueued simplices in FO order, attach a
157       * <code>Head</code> to this instance.
158       */
159      public final static class Tail {
160    
161        static final int ENTRIES_DEFAULT_LENGTH = 100;
162        Simplex[] entries;
163        int current;
164        Tail next;
165            
166        // make an empty SimplexStream.Segment with the default length
167        public Tail() {
168          entries = new Simplex[ENTRIES_DEFAULT_LENGTH];
169          current = 0;
170          next = null;
171        }
172    
173        // make an empty SimplexStream.Segment with the given length
174        public Tail(int len) {
175          entries = new Simplex[len];
176          current = 0;
177          next = null;
178        }
179    
180        public String toString() {
181          return String.format("[QT(%d:%d): {%d}%s]",
182                               current, entries.length, 
183                               System.identityHashCode(entries),
184                               (next != null)?("->"+next.toString()):"");
185        }
186    
187        public final Tail enqueue(Simplex s) {
188          if (current < entries.length) {
189            assert(next == null);
190            entries[current++] = s;
191            return this;
192          } else {
193            next = new Tail(entries.length);
194            next.entries[next.current++] = s;
195            return next;
196          }
197        }
198      }
199    
200      /**
201       * A <code>Head</code> provides the FO part of a FIFO queue when
202       * attached to a <code>Tail</code> to which simplices have been
203       * enqueued.
204       *
205       */
206      public static final class Head {
207    
208        Simplex[] entries;
209        int current;
210        Tail orig;
211    
212        // don't use
213        private Head() {
214        }
215    
216        // make a head for a Tail
217        public Head(Tail q) {
218          entries = q.entries;
219          current = q.current;
220          orig = q;
221        }
222    
223        // Duplicate a Head. Note that interating with a copy has no effect
224        // on the original, and no effect on the underlying Tail.
225        public Head copy() {
226          Head tmp = new Head();
227          tmp.entries = entries;
228          tmp.current = current;
229          tmp.orig = orig;
230          return tmp;
231        }
232    
233        public String toString() {
234          return String.format("[QH(%d:%d): {%d} (%s)]",
235                               current, entries.length, 
236                               System.identityHashCode(entries),
237                               orig.toString());
238        }
239    
240        /**
241         * Get the next entry from the underlying Queue.
242         *
243         * <p> 
244         * This only works is we don't outrun the underlying Tail.
245         * <p>
246         * @return     next Simplex in the Queue.
247         *
248         */
249        public final Simplex nextEntry() {
250          if (current == entries.length) {
251            // System.out.printf("nextEntry() for %s\n", toString());
252            Tail n = orig.next;
253            entries = n.entries;
254            current = 0;
255            orig = n;
256          }
257          return entries[current++];
258        }
259    
260        /**
261         * Compare one Head to another.
262         *
263         * <p> This is a very fast comparison that will only work if both
264         * Head instances are heads for the same Tail. 
265         * <p>
266         * @param      q  Head to compare
267         * @return     true if this is prior to q, else false.
268         *
269         */
270        public final boolean lessThan (Head q) {
271          return ((entries != q.entries) || (current < q.current));
272        }
273    
274    
275        /**
276         * Check equality of Head with another.
277         *
278         * <p> This is used only by assertions.
279         * <p>
280         * @param      q  Head to compare
281         * @return     true if this is equal to q, else false.
282         *
283         */
284        public final boolean eql(Head q) {
285          return ((entries == q.entries) || (current == q.current));
286        }
287      }

289      /**
290       * <code>Stack</code> is a multiway segmented stack for sorted storage
291       * of simplices.
292       *
293       * <p> The class <code>SimplexStream.Stack</code> provides a simple,
294       * fast, scalable implementation of SimplexStream. Instances are arrays
295       * of segmented stacks of simplices, simplices are less than,
296       * equivalent, or greater than one another (for the purposes of the
297       * Persistence computations) if and only if the corresponding indices
298       * for their segmented stacks are <, ==, or >. In addition, the next()
299       * method drops references to the segments as soon as possible, which
300       * means that the stack consumes progressively less memory as its
301       * contents are processed.
302       */
303      public static final class Stack extends SimplexStream {
304    
305        // Rather like the Queue code above, but this is even simpler --
306        // the building block of a segmented stack.
307        private static final class Segment {
308          private final Simplex[] entries;
309          private int current;
310          private Segment next;
311            
312          // don't use
313          private Segment() {
314            entries = null;
315            current = 0;
316            next = null;
317          }
318    
319          // make an empty SimplexStream.Segment with a specified length
320          private Segment(int len) {
321            entries = new Simplex[len];
322            current = 0;
323            next = null;
324          }
325    
326          public String toString() {
327            return String.format("[Seg(%d:%d): {%d}%s]",
328                                 current, entries.length, 
329                                 System.identityHashCode(entries),
330                                 (next != null)?("->"+next.toString()):"");
331          }
332    
333          private final Segment push(Simplex s) {
334            if (current < entries.length) {
335              entries[current++] = s;
336              return this;
337            } else {
338              Segment tmp = new Segment(entries.length);
339              tmp.next = this;
340              tmp.entries[tmp.current++] = s;
341              return tmp;
342            }
343          }
344        }
345    
346        private static final int STACK_SEGMENT_LENGTH = 10;
347    
348        private final Segment[] segments;
349        private final int findex_bound;
350        private final int dimension_bound;
351        private final int segment_length;
352        private int size;
353    
354        private int segments_index;
355        private int simplex_index;
356    
357        // don't use
358        private Stack() {
359          segment_length = 0;
360          segments = null;
361          dimension_bound = 0;
362          findex_bound = 0;
363          segments_index = 0;
364          simplex_index = 0;
365          size = 0;
366        }
367    
368        /**
369         * Constructor for Stack.
370         *
371         * <p>
372         * Make a multi-way stack that automatically sorts simplices.
373         * <p>
374         * @param      max_findex Maximum Filtration Index we'll use.
375         * @param      max_d Maximum dimension of a Simplex to be pushed.
376         *
377         */
378        public Stack(int max_findex, int max_d) {
379          segment_length = STACK_SEGMENT_LENGTH;
380          dimension_bound = (max_d + 1);
381          findex_bound = (max_findex + 1);
382          segments = new Segment[dimension_bound * findex_bound];
383          segments_index = 0;
384          simplex_index = 0;
385          size = 0;
386        }
387    
388        public final String toString() {
389          return String.format("[STK(%d,%d): %d/%d, si=%d/%d]",
390                               dimension_bound, findex_bound, 
391                               segments_index, segments.length,
392                               simplex_index, segment_length);
393        }
394    
395        private final int simplexIndex(Simplex s) {
396          assert((s.findex() < findex_bound) && (s.dimension() < dimension_bound));
397          return (s.findex() * dimension_bound) + s.dimension();
398        }
399    
400        /**
401         * Push a Simplex into the stream.
402         *
403         * <p> Add the Simplex s to the stream so that it will be retrieved
404         * in the proper order. 
405         * <p>
406         * @param      s  Simplex instance to push.
407         * @return     s
408         *
409         */
410        public final Simplex push(Simplex s) {
411          int index = simplexIndex(s);
412          size++;
413          if (segments[index] == null) 
414            segments[index] = new Segment(segment_length);
415          segments[index] = (segments[index]).push(s);
416          return s;
417        }
418    
419        /**
420         * How many simplices are in the stream?
421         *
422         * <p>
423         * @return     The number of simplices in the stream.
424         *
425         */
426        public final int size() {
427          return size;
428        }
429    
430        /**
431         * Max dimension of simplices in the stream.
432         *
433         * <p>
434         * @return     Max dimension of simplices in the stream.
435         *
436         */
437        public final int maxDimension() {
438          return (dimension_bound - 1);
439        }
440    
441        /**
442         * Is there a next Simplex in the stream?
443         *
444         * <p>
445         * @return     Return true if the stream is not yet empty.
446         *
447         */
448        public boolean hasNext() {
449          if (segments[segments_index] != null) {
450            if (simplex_index < segments[segments_index].current)
451              return true;
452            else {
453              simplex_index = 0;
454              Segment next = segments[segments_index].next;
455              segments[segments_index] = next;
456              if (next != null) {
457                assert(0 < next.current);
458                return true;
459              }
460            }
461          }
462    
463          assert(segments[segments_index] == null);
464          assert(simplex_index == 0);
465    
466          while ((segments_index < segments.length) && 
467                 (segments[segments_index] == null))
468            segments_index++;
469    
470          if (segments_index == segments.length)
471            return false;
472    
473          assert(segments[segments_index] != null);
474          assert(0 < segments[segments_index].current);
475          return true;
476        }
477    
478        /**
479         * Return the next Simplex in the stream. This operations
480         * <em>removes</em> the simplex returned from the stream, so
481         * repeated iterations are impossible with the method.
482         *
483         * <p>
484         * @return     The smallest remaining Simplex instance.
485         *
486         */
487        public Simplex next() {
488          Segment seg = segments[segments_index];
489          Simplex s;
490          if (seg != null) {
491            if (simplex_index < seg.current) {
492              s = seg.entries[simplex_index];
493              seg.entries[simplex_index++] = null;
494              assert(size > 0);
495              size--;
496              return s;
497            } else {
498              simplex_index = 0;
499              seg = segments[segments_index] = segments[segments_index].next;
500              if (seg != null) {
501                assert(0 < seg.current);
502                s = seg.entries[simplex_index];
503                seg.entries[simplex_index++] = null;
504                assert(size > 0);
505                size--;
506                return s;
507              }
508            }
509          }
510    
511          assert(segments[segments_index] == null);
512          assert(simplex_index == 0);
513    
514          while ((segments_index < segments.length) && 
515                 ((seg = segments[segments_index]) == null))
516            segments_index++;
517    
518          if (segments_index == segments.length)
519            return null;
520    
521          assert(segments[segments_index] != null);
522          assert(0 < seg.current);
523          s = seg.entries[simplex_index];
524          seg.entries[simplex_index++] = null;
525          assert(size > 0);
526          size--;
527          return s;
528        }

530        /**
531         * Instances provide Iterator<Simplex> for non-destructive
532         * iterating over Stack entries. That is, repeated iteration over a
533         * Stack instance is possible with multiple instances of this
534         * class.
535         */
536        private static final class StackIterator implements Iterator<Simplex> {
537          private Segment[] segments;
538          private int segments_index;
539          private Segment current_seg;
540          private int simplex_index;
541          private int fixed_dimension;
542          private final int dimension_bound;
543    
544          private StackIterator() {
545            segments = null;
546            segments_index = 0;
547            current_seg = null;
548            simplex_index = 0;
549            fixed_dimension = -1;
550            dimension_bound = 0;
551          }
552    
553          public StackIterator(Stack stack) {
554            segments = stack.segments;
555            segments_index = 0;
556            current_seg = null;
557            simplex_index = 0;
558            fixed_dimension = -1;
559            dimension_bound = stack.dimension_bound;
560          }
561    
562          public StackIterator(Stack stack, int dimension) {
563            segments = stack.segments;
564            segments_index = -1;
565            current_seg = null;
566            simplex_index = 0;
567            if ((dimension < 0) || (dimension >= stack.dimension_bound))
568              throw new IllegalArgumentException(dimension + 
569                                                 " must be >= 0 and <= " + 
570                                                 stack.dimension_bound); 
571            fixed_dimension = dimension;
572            dimension_bound = stack.dimension_bound;
573          }
574    
575          /**
576           * Returns <tt>true</tt> if the iterator has more simplices.
577           *
578           * @return <tt>true</tt> if the iterator has more simplices, else
579           * <tt>false</tt>.
580           */
581          public final boolean hasNext() {
582            while(true) {
583              while (current_seg == null) {
584                if (segments_index >= segments.length)
585                  return false;
586                else {
587                  if (fixed_dimension >= 0) {
588                    if (segments_index < 0)
589                      segments_index = fixed_dimension;
590                    else
591                      segments_index += dimension_bound;
592                    if (segments_index >= segments.length)
593                      return false;
594                    else
595                      current_seg = segments[segments_index];
596                  } else
597                    current_seg = segments[segments_index++];
598                }
599              }
600              while (current_seg != null) {
601                while ((simplex_index < current_seg.current) &&
602                       (current_seg.entries[simplex_index] == null))
603                  simplex_index++;
604                if (simplex_index < current_seg.current)
605                  return true;
606                else {
607                  simplex_index = 0;
608                  current_seg = current_seg.next;
609                }
610              }
611            }
612          }
613    
614          /**
615           * Returns the next Simplex in the iteration of the Stack.
616           * Used in concert with the {@link #hasNext()} method returns
617           * return each Simplex in the Stack exactly once. Does
618           * <em>not</em> have any side effects on the stream.
619           * <p>
620           * @return the next Simplex in the Stack.
621           * @exception NoSuchElementException Stack has no more elements.
622           */
623          public final Simplex next() {
624            while(true) {
625              while (current_seg == null) {
626                if (segments_index >= segments.length)
627                  throw new NoSuchElementException();
628                else {
629                  if (fixed_dimension >= 0) {
630                    if (segments_index < 0)
631                      segments_index = fixed_dimension;
632                    else
633                      segments_index += dimension_bound;
634                    if (segments_index >= segments.length)
635                      throw new NoSuchElementException();
636                    else
637                      current_seg = segments[segments_index];
638                  } else
639                    current_seg = segments[segments_index++];
640                }
641              }
642              while (current_seg != null) {
643                while ((simplex_index < current_seg.current) &&
644                       (current_seg.entries[simplex_index] == null))
645                  simplex_index++;
646                if (simplex_index < current_seg.current) {
647                  return current_seg.entries[simplex_index++];
648                } else {
649                  simplex_index = 0;
650                  current_seg = current_seg.next;
651                }
652              }
653            }
654          }
655    
656          /**
657           * Unsupported remove() operation.
658           *
659           * @exception UnsupportedOperationException 
660           */
661          public void remove() {
662            throw new UnsupportedOperationException();
663          }
664        }
665    
666        /**
667         * Make a non-destructive iterator for the Stack.
668         * <p>
669         *
670         * @return  Iterator<Simplex> instance for the stack.
671         *
672         * @see        java.util.Iterator
673         */
674        public Iterator<Simplex> iterator() {
675          return new StackIterator(this);
676        }
677    
678        /**
679         * Make a non-destructive iterator for fixed dimensional entries in the Stack.
680         * <p>
681         *
682         * @param      d  dimension of entries we want to iterate over
683         * @return  Iterator<Simplex> instance for the stack.
684         *
685         * @see        java.util.Iterator
686         */
687        public Iterator<Simplex> iterator(int d) {
688          return new StackIterator(this, d);
689        }
690      }
