025    package edu.stanford.math.plex;
026    import static java.lang.Math.*;
027    
028    /**
029     * A <code>Torus</code> instance is  discrete simulation of a d-dimensional torus. For testing.
030     *
031     * <p>Simplest implementation of a d-dimensional torus. Intended for testing.
032     *
033     * @version $Id: Torus.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
034     */
035    
036    public final class Torus extends PointData.NSpace {
037    
038      static final double COORD_LIMIT = 2.0 * PI;
039    
040      final double[] points;
041      final int dimensions;
042    
043      /**
044       * The number of dimensions of the space.
045       *
046       * <p>
047       *
048       * @return     the dimension of the data.
049       */
050      public final int dimension() {
051        return dimensions;
052      }
053       
054      /**
055       * i-th coordinate of point p;
056       *
057       * <p>
058       * @param      p   the point
059       * @param      i   the coordinate index
060       * @return     i-th coordinate
061       */
062      public double coordinate(int p, int i) {
063        int p_base = dimensions * p;
064        if ((i % 1) == 1) 
065          return sin(points[p_base + i]);
066        else
067          return cos(points[p_base + i]);
068      }
069    
070      /**
071       * The number of data points.
072       *
073       * <p>
074       *
075       * @return     the number of points
076       */
077      public final int count() {
078        if (points != null)
079          return (points.length/dimensions)-1;
080        else
081          return 0;
082      }
083       
084      /**
085       * Toroidal distance.
086       *
087       * <p>
088       *
089       * @param      p1   the first point
090       * @param      p2   the second point
091       * @return     the distance between p1 and p2 on the torus.
092       */
093      public final double distance (int p1, int p2) {
094        if (p1 == p2)
095          return 0.0;
096        else if ((p1 == 0) || (p2 == 0))
097          return Double.MAX_VALUE;
098        else {
099          int p1_base = dimensions * p1;
100          int p2_base = dimensions * p2;
101          double accum = 0.0;
102          for (int i = 0; i < dimensions; i++) {
103            double p1ix = cos(points[p1_base + i]);
104            double p1iy = sin(points[p1_base + i]);
105            double p2ix = cos(points[p2_base + i]);
106            double p2iy = sin(points[p2_base + i]);
107            accum  += (((p1ix - p2ix) * (p1ix - p2ix)) + 
108                       ((p1iy - p2iy) * (p1iy - p2iy)));
109          }
110          return sqrt(accum);
111        }
112      }
113    
114      // Make sure that the data and dimension arguments are not absurd.
115      private boolean pointsAreConsistent(double[] data, int dimension) {
116        if (dimension < 1)
117          return false;
118        if ((data.length % dimension) != 0)
119          return false;
120        if (data.length <= dimension)
121          return false;
122        for (int i = 0; i < dimension; i++) {
123          if (data[i] != 0.0)
124            return false;
125        }
126        return true;
127      }
128    
129      /**
130       * Make a Torus with a given data array.
131       *
132       * <p>
133       *
134       * @param     data double[] of point data.
135       */
136      public Torus(double[] data, int d) {
137        points = data;
138        dimensions = d;
139        assert pointsAreConsistent(data, dimensions);
140      }
141    
142      // exponentiation of integers
143      private final int expt(int n, int k) {
144        assert (n >= 0) && (k >= 0);
145        if (k == 0) {
146          return 1;
147        } else if (n == 0) {
148          return 0;
149        } else {
150          int accum = 1;
151          for (int i = 0; i < k; i++) {
152            assert (accum > 0) && ((n * accum) > 0);
153            accum = n * accum;
154          }
155          return accum;
156        }
157      }
158    
159      // Increment a d-dimensional array of points so that we systematically
160      // go through all of the points in an incr-length grid.
161      private final boolean incr_pt(double[] pt, double incr) {
162        for (int i = 0; i < pt.length; i++) {
163          if ((incr + pt[i]) < COORD_LIMIT) {
164            pt[i] += incr; 
165            // clear the previous indices
166            while(i > 0) pt[--i] = 0.0;
167            return true;
168          } 
169        }
170        return false;
171      }
172    
173      /**
174       * Make an n-way grid on the unit Torus.
175       *
176       * <p>
177       *
178       * @param      n   grid size parameter
179       */
180      public Torus(int n, int d) {
181        assert (n > 0) && (d > 0);
182        int pt_counter = expt(n, d);
183        double[] data = new double[d * (pt_counter + 1)];
184        int counter = d;
185        double[] pt = new double[d];
186        double incr = COORD_LIMIT/((double) n);
187        for (int i = 1; i <= pt_counter; i++) {
188          for (int j = 0; j < d; j++) {
189            data[(d * i) + j] = pt[j];
190          }
191          boolean success = incr_pt(pt, incr);
192          assert (i == pt_counter) || success;
193        }
194        assert(pointsAreConsistent(data, d));
195        points = data;
196        dimensions = d;
197      }
198    
199      // Argless constructor for this class Test use only.
200      public Torus() {
201        double[] data = new double[] {0, 0, 0, 0, .5, .5, 0, .5, .5, 0, 
202                                      .25, 0, 0, .25, .25, .25, .25, .5};
203        dimensions = 2;
204        assert(pointsAreConsistent(data, dimensions));
205        points = data;
206      }
207    }
