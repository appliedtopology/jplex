024    package edu.stanford.math.plex;
025    
026    import java.util.*;
027    
028    /**
029     * The <code>DistanceData</code> class is the simplest implementation
030     * of NSpace. Intended for smallish data sets.
031     *
032     * @version $Id: DistanceData.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
033     */
034    
035    public final class DistanceData extends PointData {
036    
037      final double[][] d;
038      final int numPoints;
039    
040      /**
041       * The number of data points.
042       *
043       * <p>
044       *
045       * @return     the number of points
046       */
047      public final int count() {
048        if (d != null)
049          return (d.length);
050        else
051          return 0;
052      }
053       
054      /**
055       * Euclidean distance.
056       *
057       * <p>
058       *
059       * @param      p1   the first point
060       * @param      p2   the second point
061       * @return     the distance between p1 and p2.
062       */
063      public final double distance (int p1, int p2) {
064        if (p1 == p2)
065          return 0.0;
066        else {
067          return d[p1-1][p2-1];
068        }
069    
070      }
071    
072      // Constructor for this class.
073      public DistanceData(double[][] data) {
074        super();
075        d = data;
076        numPoints = d.length;
077      }
078    
079    
080    }
