046      // Compute the size of the intersection of v1 and v2, assuming that no
047      // elements appear in either v1 or v2 more than once. This method has the
048      // advantage that it doesn't need the arguments to sorted. If the sets
049      // are already sorted or we actually want the intersection, see below.
050      public static final int intersection_count(int[] v1, int[] v2) {
051        int[] tmp = new int[v1.length + v2.length];
052        if (tmp.length == 0)
053          return 0;
054        int tmp_counter = 0;
055        for (int i = 0; i < v1.length; i++)
056          tmp[tmp_counter++] = v1[i];
057        for (int i = 0; i < v2.length; i++)
058          tmp[tmp_counter++] = v2[i];
059        assert(tmp_counter == tmp.length);
060        Arrays.sort(tmp);
061    
062        int intersection_counter = 0;
063        int current_tmpval = tmp[0];
064        int current_hits = 1;
065        for (int i = 1; i < tmp.length; i++) {
066          int x = tmp[i];
067          if (x == current_tmpval)
068            current_hits++;
069          else {
070            if (current_hits > 1) {
071              assert(current_hits == 2);
072              intersection_counter++;
073            }
074            current_hits = 1;
075            current_tmpval = x;
076          }
077        }
078          
079        return intersection_counter;
080      }

082      /**
083       * Compute the overlap size for two sorted arrays using the zipper algorithm.
084       *
085       * @param      v1  First sorted array.
086       * @param      v2  Second sorted array.
087       * @return     The number of common elements of v1 and v2.
088       *
089       */
090      public final static int sorted_intersection_count(int[] v1, int[] v2) {
091        int count = 0;
092    
093        int i = 0;
094        int i_max = v1.length-1;
095    
096        int j = 0;
097        int j_max = v2.length-1;
098    
099        if ((i > i_max) || (j > j_max))
100          return count;
101    
102        int max_xi = v1[i_max];
103        int max_xj = v2[j_max];
104    
105        int xi = v1[i];
106        if (xi > max_xj) return count;
107        int xj = v2[j];
108        if (xj > max_xi) return count;
109    
110        while (true) {
111          if (xi == xj) {
112            count++;
113            i++;
114            j++;
115            if ((i > i_max) || (j > j_max))
116              return count;
117            else {
118              xi = v1[i];
119              if (xi > max_xj) return count;
120              xj = v2[j];
121              if (xj > max_xi) return count;
122            }
123          } else if (xi < xj) {
124            i++;
125            xi = v1[i];
126            if (xi > max_xj) return count;
127          } else {
128            j++;
129            xj = v2[j];
130            if (xj > max_xi) return count;
131          }
132        }
133      }

136      /**
137       * Take subset_size distinct entries from initial segment of length
138       * set_size in set[] and put them in the initial segment of subset[]. The
139       * ordering of the entries is preserved -- that is, if set[i] and
140       * set[i+d] both go into subset, the index of set[i] is less than that of
141       * set[i+d].  <p>
142       * @param      set an int[] from which to select.
143       * @param      set_size  an int specifying how many entries to use.
144       * @param      subset an int[] in which to deposit the selections.
145       * @param      subset_size an int specifying how many entries to select.
146       */
147      public static void random_subset(int[] set, int set_size, 
148                                       int[] subset, int subset_size) {
149        if (subset_size > set_size)
150          throw new IllegalArgumentException("Requested subset is larger than argument set.");
151        if (subset_size > subset.length)
152          throw new IllegalArgumentException("Requested subset is larger than storage given.");
153        if (set_size > set.length)
154          throw new IllegalArgumentException("Set storage smaller than argument size.");
155        if (subset_size <= 0)
156          return;
157    
158        int[] tmp = new int[set_size];
159        random_permutation(tmp);
160        for (int i = 0; i < subset_size; i++) {
161          subset[i] = tmp[i];
162        }
163        Arrays.sort(subset, 0, subset_size);
164        for (int i = 0; i < subset_size; i++) {
165          subset[i] = set[subset[i]];
166        }
167      }

213      // confirm that this is a permutation
214      private static boolean is_permutation (int[] perm) {
215        int[] tmp = new int[perm.length];
216        for (int i = 0; i < perm.length; i++)
217          tmp[perm[i]]++;
218        for (int i = 0; i < perm.length; i++) {
219          if (tmp[i] != 1)
220            return false;
221        }
222        return true;
223      }

225      /**
226       * Make a random permutation of the integers 0 through perm.length-1
227       * and put them in perm.
228       * <p>
229       * @param      perm  an int[] to hold a random permutation.
230       */
231      public static void random_permutation(int[] perm) {
232        if (perm.length <= 0)
233          return;
234        // start with the identity
235        for (int i = 0; i < perm.length; i++)
236          perm[i] = i;
237        for (int current = perm.length; current > 0; current--) {
238          int swap_to = rand.nextInt(current);
239          int swap_from = current - 1;
240          int tmp = perm[swap_to];
241          perm[swap_to] = perm[swap_from];
242          perm[swap_from] = tmp;
243        }
244        assert(is_permutation(perm));
245      }

438      /**
439       * Take subset_size distinct entries from the integers 0 through N-1,
440       * and put them in the initial segment of subset[].
441       * <p>
442       * @param      N  an int specifying the range from which to choose the subset.
443       * @param      subset an int[] in which to deposit the selections.
444       * @param      subset_size  an int specifying how many entries to use.
445       */
446      public static void random_index_subset(int N, int[] subset, int subset_size) {
447        if (subset_size > subset.length)
448          throw new IllegalArgumentException("Requested subset is larger than storage given.");
449        if (subset.length <= 0)
450          return;
451    
452        int min_value = subset_size-1;
453        int max_value = N-1;
454        int current_index = subset_size - 1;
455        int select_value = min_value;
456        while (current_index >= 0) {
457          if (max_value > min_value) 
458            select_value = min_value + rand.nextInt(1 + (max_value - min_value));
459          else 
460            select_value = min_value;
461          subset[current_index] = select_value;
462          assert((select_value >= 0) && (select_value <= N-1));
463          assert((current_index >= (subset_size-1)) || 
464                 (subset[current_index] < subset[current_index+1]));
465          current_index--;
466          max_value = select_value-1;
467          min_value--;
468        }
469        assert(current_index == -1);
470      }

472      // Compare two int[] arguments, returning negative if less, 0 if equal,
473      // and positive if greater so that we can use it in binary search.
474      static final int comparePtArrays (int[] pts1, int[] pts2) {
475        int len1 = pts1.length;
476        int len2 = pts2.length;
477        int count = (len2 > len1)?len2:len1;
478        for (int i = 0; i < count; i++) {
479          if (pts1[i] != pts2[i])
480            return (pts1[i] - pts2[i]);
481        }
482        return (len1 - len2);
483      }

485      // Compare two int[] arguments, returning TRUE if equal, else FALSE.
486      static final boolean equalPtArrays (int[] pts1, int[] pts2) {
487        if (pts1 == pts2)
488          return true;
489        else if ((pts1 == null) || (pts2 == null))
490          return false;
491        else if (pts1.length != pts2.length)
492          return false;
493        else {
494          for (int i = 0; i < pts1.length; i++) {
495            if (pts1[i] != pts2[i])
496              return false;
497          }
498          return true;
499        }
500      }

502      // Find the first index i such that arg >= pts[i]. Returns -1 for failure.
503      // We start from argument "lo", which is 
504      private static final int firstSmallerPtIndex (int arg, int lo, int[] pts) {
505        // Binary search.
506        int hi = pts.length - 1;
507    
508        if ((lo > hi) || (pts[hi] > arg))
509          return -1;
510        else if (arg >= pts[lo])
511          return lo;
512        else {
513          // Do a binary search. At the top of the loop, we always have
514          // that (pts[lo] > arg), lo != hi, and (arg >= pts[hi]).
515          while (lo < hi) {
516            int mid = (lo + hi)/2;
517    
518            assert(lo < hi);
519            assert(pts[lo] > arg);
520            assert(pts[hi] <= arg);
521    
522            if (mid == lo)
523              return hi;
524            else if (pts[mid] > arg)
525              lo = mid;
526            else
527              hi = mid;
528          }
529          // should never get here
530          assert(false);
531          return -1;
532        }
533      }

535      // Returns TRUE if ipts is a subset of jpts.
536      private static final boolean ptArrayIsSubset (int[] ipts, int[] jpts) {
537        int j = 0;
538    
539        for (int i = 0; i < ipts.length; i++) {
540          if ((ipts.length - jpts.length) > (i - j))
541            // this the same as (ipts.length-i) > (jpts.length-j), which
542            // would mean that we have more points to find than we have
543            // places to look for them
544            return false;
545    
546          j = firstSmallerPtIndex (ipts[i], j, jpts);
547          if ((j < 0) || (ipts[i] != jpts[j]))
548            // jpts is decreasing, so if the first entry is too small,
549            // we'll never find a match
550            return false;
551          else
552            j++;            
553        }
554      
555        return true;
556      }

559      /**
560       * An array of Betti numbers (int[]) boxed up for convenience.
561       */
562      public static class BettiNumbers {
563        private final int[] numbers;
564            
565        // make sure the numbers array is okay.
566        static private boolean init_okay(int[] init) {
567          for (int i = 0; i < init.length; i++) {
568            if (init[i] < 0)
569              return false;
570          }
571          if ((init.length == 0) || (init[init.length-1] == 0))
572            return false;
573          return true;
574        }
575    
576        // unused
577        private BettiNumbers() { numbers = null; }
578    
579        /**
580         * Encapsulate a verified array of Betti numbers.
581         *
582         * <p>
583         * @param      init an int[] of betti numbers.
584         * @exception  IllegalArgumentException 
585         */        
586        public BettiNumbers(int[] init) {
587          if (!init_okay(init))
588            throw new IllegalArgumentException("Bad betti numbers");
589          numbers = new int[init.length];
590          for(int i = 0; i < init.length; i++)
591            numbers[i] = init[i];
592        }
593    
594        /**
595         * Make BettiNumbers printable.
596         *
597         * <p>
598         * @return      String form of an array of Betti numbers.
599         */        
600        public String toString() {
601          String base = "BN{";
602          for (int i = 0; i < numbers.length-1; i++) {
603            base = base + String.format("%d, ", numbers[i]);
604          }
605          base = base + String.format("%d}", numbers[numbers.length-1]);
606          return base;
607        }
608    
609        /**
610         * Determine if two instances of BettiNumbers are equal.
611         *
612         * <p>
613         * param  obj    Object to check equality against.
614         * @return      True if equal, else false.
615         */        
616        public boolean equals(Object obj) {
617          if (!(obj instanceof BettiNumbers))
618            return false;
619          BettiNumbers b = (BettiNumbers) obj;
620          return Plex.equalPtArrays(numbers, b.numbers);
621        }
622      }

624      /**
625       * Summarize the semi-infinite PersistenceIntervals.
626       *
627       * <p>
628       * @param      coll  A collection of PersistenceIntervals.
629       * @return     An instance of Plex.BettiNumbers.
630       */
631      public static BettiNumbers FilterInfinite(PersistenceInterval[] coll) {
632        int max_dimension = 0;
633        for (PersistenceInterval i : coll) {
634          if (i.infiniteExtent() && (i.dimension > max_dimension))
635            max_dimension = i.dimension;
636        }
637        int[] results = new int[max_dimension+1];
638        for (PersistenceInterval i : coll) {
639          if (i.infiniteExtent())
640            results[i.dimension]++;
641        }
642        return new BettiNumbers(results);
643      }

830      // Get rid of really short PersistenceInterval.Float instances.
831      private static PersistenceInterval.Float[] filter_pintervals
832        (PersistenceInterval.Float[] vec, double delta) {
833        int return_len = 0;
834        for (int i = 0; i < vec.length; i++) {
835          PersistenceInterval.Float p = vec[i];
836          if (Math.abs(p.start - p.start) > delta)
837            return_len++;
838        }
839        PersistenceInterval.Float[] return_value = 
840          new PersistenceInterval.Float[return_len];
841        return_len = 0;
842        for (int i = 0; i < vec.length; i++) {
843          PersistenceInterval.Float p = vec[i];
844          if (Math.abs(p.start - p.start) > delta)
845            return_value[return_len++] = vec[i];
846        }
847        return return_value;
848      }

850      /**
851       * Are two PersistenceInterval.Float[] arguments sufficiently equal?
852       *
853       * <p>
854       * @param      arg1 PersistenceInterval.Float[].
855       * @param      arg2 PersistenceInterval.Float[].
856       * @param      delta double
857       * @return     true if the arguments are equivalent up the precision of the stream and delta.
858       *
859       */
860      public static boolean equalPersistenceIntervals(PersistenceInterval.Float[] arg1, 
861                                                      PersistenceInterval.Float[] arg2, 
862                                                      double delta) {
863        if (arg1 == arg2)
864          return true; 
865        else {
866          arg1 = filter_pintervals(arg1, delta);
867          arg2 = filter_pintervals(arg2, delta);
868          if (arg1.length != arg2.length)
869            return false;
870          for (int i = 0; i < arg1.length; i++) {
871            PersistenceInterval.Float p1 = arg1[i];
872            PersistenceInterval.Float p2 = arg2[i];
873            if ((Math.abs(p1.start - p2.start) > 1.5 * delta) ||
874                (Math.abs(p1.end - p2.end) > 1.5 * delta))
875              return false;
876          }
877          return true;
878        }
879      }
