045       * Tests some behavior.
046       *
047       * @exception  whatever
048       * @throws what is the difference between these 2
049       *
050       * @see        java.lang.System#getProperty(java.lang.String)
051       * @see        SecurityManager#checkPermission
052       */    
053      @Test
054        public void testSomeBehavior() {
055        PointData rdata = (PointData) new EuclideanArrayData(20, 4);
056        RipsStream zstr = new RipsStream(0.0, 2.0, 2, rdata);
057        SimplexStream zstream = (SimplexStream) zstr;
058        ExplicitStream estream = Plex.makeExplicit(zstr);
059        assertEquals("converted right number of simplices", zstr.size(), estream.size());
060        ExplicitStream.DComplex points = estream.dump(0);
061        int counter = (points.C()).length;
062        while (points.hasNext()) {
063          points = points.next();
064          counter += (points.C()).length;
065        }
066        assertEquals("counted right number of simplices", zstr.size(), counter);
067        ExplicitStream.DComplex triangles = estream.dump(2);
068        int[][] tvec = triangles.C();
069        int[] s2 = tvec[tvec.length/2];
070        int[] l1 = new int[] {s2[0], s2[1]};
071        estream.remove(l1);
072        boolean caught_the_ise = false;
073        assertFalse("verification fails", (estream.verify(false)));
074          try{ estream.close(); } 
075          catch(IllegalStateException ise) { 
076            caught_the_ise = true;
077            estream.ensure_all_faces();        
078          }
079          finally{}
080        assertTrue("catch occurred", caught_the_ise);
081        assertTrue("verification succeeds after catch", (estream.verify(false)));
082        estream.close();
083    
084        // check iterator and stream next equivalence
085        {
086          Iterator<Simplex> iterator = estream.iterator();
087          int last_fi = -1;
088          int last_dimension = -1;
089          Simplex iterator_current = null;
090          Simplex stream_current = null;
091    
092          assertTrue("iterator start check", iterator.hasNext());
093          assertTrue("iterator start check", estream.hasNext());
094          while(iterator.hasNext()) {
095            iterator_current = iterator.next();
096            assertTrue("iterator check", (estream.hasNext() &&
097                                          ((stream_current = estream.next()) != null) &&
098                                          (iterator_current.equals(stream_current))));
099            if (iterator_current.dimension() > last_dimension) {
100              last_dimension = iterator_current.dimension();
101              last_fi = -1;
102            }
103            assertTrue("nondecreasing fi's", 
104                       (last_fi <= iterator_current.findex()));
105            last_fi = iterator_current.findex();
106          }
107          assertTrue("estream iterator end check", !estream.hasNext());
108        }
109    
110        ExplicitStream estream2 = Plex.ExplicitStream();
111        estream2.add(new double[][] { {2.1, 3.1}, {1.4, 2.3}, {2.9, .99} }, 
112                     new double[] {0.0, 0.0, 1.0});
113        estream2.add(new int[][] { {1}, {3}, {2} }, 
114                     new double[] {0.0, 0.0, 0.0});
115        ExplicitStream.DComplex force_close = estream.dump(1);
116      }
