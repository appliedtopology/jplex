063      // Some not very precise timing information -- a very simple-minded
064      // test suggested that Simplex creation was within a factor of 2 of a
065      // handcrafted C code implementation. The numbers here are only
066      // interesting as relative values, because they were collected without
067      // trying for the best performance. I think that we can probably halve
068      // these times when running in production mode.
069      //
070      // Construction only: 8.5M simplices Time: 3.6 secs
071      // Construction and interning of 8.5M simplices -- Time: 6.8 secs
072      // Construction, interning, bdy operation on 8.5M -- Time: 19.4 secs
073      // Simplex construction -- Chain construction on 8.5M -- Time: 22.5 secs
074      // Simplex construction -- add chains together on 8.5M -- Time: 25.8 secs

076      /**
077       * Tests creating a RipsStream and interning the elements in it, as
078       * well as doing some Chain arithmetic.
079       *
080       */    
081      @Test
082        public void testSomeBehavior() {
083        PointData rdata = (PointData) new EuclideanArrayData(40, 4); // EuclideanArrayData(120, 4);
084        int rstr_size = 0; 
085        int rtbl_size = 0;
086        Simplex s, t;
087        RipsStream rstr = new RipsStream(.05, 2.0, 3, rdata);
088        int rstr_initial_size = rstr.size();
089        SimplexTable rtbl = new SimplexTable(1000);
090        int initial_limit = rtbl.limit();
091        Iterator<Simplex> riter = rstr.iterator();
092        while ((riter.hasNext() && ((s = riter.next()) != null)) && 
093               (riter.hasNext() && ((t = riter.next()) != null))) {
094          Simplex[] b = s.boundary(); 
095          Simplex[] c = t.boundary(); 
096          if (b != null) 
097            for (int i = 0; i < b.length; i++) b[i] = rtbl.get(b[i]);
098          rtbl.put(s);
099          if (c != null) 
100            for (int i = 0; i < c.length; i++) c[i] = rtbl.get(c[i]);
101          rtbl.put(t);
102          Chain d = Chain.fromBoundary(b, Persistence.baseModulus());
103          Chain e = Chain.fromBoundary(c, Persistence.baseModulus());
104          if(d != null) 
105            assert(rtbl.get(d.maxS()) != null);
106          if(e != null) {
107            if (d != null)
108              e = e.add(d, 1);
109            assert((e == null) || (rtbl.get(e.maxS()) != null));
110          }
111        }
112        assertTrue("table big enough", ((rtbl.size() >= (rstr_initial_size - 2)) && 
113                                        (rtbl.size() > 0)));
114        assertTrue("table grew", (rtbl.limit() > initial_limit));
115        int found = 0;
116        while (((s = rstr.next()) != null) && ((t = rstr.next()) != null)) {
117          assertTrue("found s", rtbl.get(s) != null);
118          assertTrue("found t", rtbl.get(t) != null);
119          found += 2;
120        }
121        assertEquals("found all", rtbl.size(), found);
122        assertEquals("stream empty", rstr.size(), 0);
123      }
