064      /**
065       * Tests some behavior.
066       *
067       * @exception  whatever
068       * @throws what is the difference between these 2
069       *
070       * @see        java.lang.System#getProperty(java.lang.String)
071       * @see        SecurityManager#checkPermission
072       */    
073      @Test
074        public void testSomeBehavior() {
075        SimplexStream stream = TmpStream.PaperTestCase();
076        Persistence P = new Persistence();
077        PersistenceInterval[] expected = new PersistenceInterval[5];
078    
079        expected[0] = new PersistenceInterval.Int(0, 0, 1);
080        expected[1] = new PersistenceInterval.Int(0, 0);
081        //expected[2] = new PersistenceInterval.Int(0, 1, 1);
082        expected[2] = new PersistenceInterval.Int(0, 1, 2);
083        expected[3] = new PersistenceInterval.Int(1, 2, 5);
084        expected[4] = new PersistenceInterval.Int(1, 3, 4);
085    
086        PersistenceInterval[] intervals = P.computeRawIntervals(stream, false, 7);
087        int counter = 0;
088        for (PersistenceInterval i : intervals) {
089          assertTrue("Got expected results", i.equals(expected[counter++]));
090        }
091    
092        // make sure that the homology of the d-sphere is what we expect
093        for (int d = 1; d <= 5; d++) {
094          SimplexStream ds = PointData.Discrete.DSphere(d);
095          Persistence p = Plex.Persistence();
096          Plex.BettiNumbers gen = 
097            Plex.FilterInfinite(p.computeIntervals(ds));
098          int[] exp_init = new int[d+1];
099          exp_init[0] = 1;
100          exp_init[d] = 1;
101          Plex.BettiNumbers exp_bn = new Plex.BettiNumbers(exp_init);
102          assertEquals("sphere H* generators", exp_bn, gen);
103        }
104    
105        // Make sure that the homology of the S(d,k) (the k-skeleton of the
106        // d-sphere) is what we expect.
107        {
108          SimplexStream ds;
109          Persistence p;
110          Plex.BettiNumbers gen_dk;
111    
112          p = Plex.Persistence();
113          ds = PointData.Discrete.DSphereKskeleton(3,1);
114          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
115          assertEquals("S(d,k) homology", gen_dk, 
116                       new Plex.BettiNumbers(new int[] {1, 6}));
117    
118          p = Plex.Persistence();
119          ds = PointData.Discrete.DSphereKskeleton(3,2);
120          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
121          assertEquals("S(d,k) homology", gen_dk, 
122                       new Plex.BettiNumbers(new int[] {1, 0, 4}));
123    
124          p = Plex.Persistence();
125          ds = PointData.Discrete.DSphereKskeleton(4,1);
126          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
127          assertEquals("S(d,k) homology", gen_dk, 
128                       new Plex.BettiNumbers(new int[] {1, 10}));
129    
130          p = Plex.Persistence();
131          ds = PointData.Discrete.DSphereKskeleton(4,2);
132          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
133          assertEquals("S(d,k) homology", gen_dk, 
134                       new Plex.BettiNumbers(new int[] {1, 0, 10}));
135    
136          p = Plex.Persistence();
137          ds = PointData.Discrete.DSphereKskeleton(4,3);
138          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
139          assertEquals("S(d,k) homology", gen_dk, 
140                       new Plex.BettiNumbers(new int[] {1, 0, 0, 5}));
141    
142          p = Plex.Persistence();
143          ds = PointData.Discrete.DSphereKskeleton(5,1);
144          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
145          assertEquals("S(d,k) homology", gen_dk, 
146                       new Plex.BettiNumbers(new int[] {1, 15}));
147    
148          p = Plex.Persistence();
149          ds = PointData.Discrete.DSphereKskeleton(5,2);
150          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
151          assertEquals("S(d,k) homology", gen_dk, 
152                       new Plex.BettiNumbers(new int[] {1, 0, 20}));
153    
154          p = Plex.Persistence();
155          ds = PointData.Discrete.DSphereKskeleton(5,3);
156          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
157          assertEquals("S(d,k) homology", gen_dk, 
158                       new Plex.BettiNumbers(new int[] {1, 0, 0, 15}));
159    
160          p = Plex.Persistence();
161          ds = PointData.Discrete.DSphereKskeleton(5,4);
162          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
163          assertEquals("S(d,k) homology", gen_dk, 
164                       new Plex.BettiNumbers(new int[] {1, 0, 0, 0, 6}));
165    
166          p = Plex.Persistence();
167          ds = PointData.Discrete.DSphereKskeleton(6,1);
168          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
169          assertEquals("S(d,k) homology", gen_dk, 
170                       new Plex.BettiNumbers(new int[] {1, 21}));
171    
172          p = Plex.Persistence();
173          ds = PointData.Discrete.DSphereKskeleton(6,2);
174          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
175          assertEquals("S(d,k) homology", gen_dk, 
176                       new Plex.BettiNumbers(new int[] {1, 0, 35}));
177    
178          p = Plex.Persistence();
179          ds = PointData.Discrete.DSphereKskeleton(6,3);
180          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
181          assertEquals("S(d,k) homology", gen_dk, 
182                       new Plex.BettiNumbers(new int[] {1, 0, 0, 35}));
183    
184          p = Plex.Persistence();
185          ds = PointData.Discrete.DSphereKskeleton(6,4);
186          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
187          assertEquals("S(d,k) homology", gen_dk, 
188                       new Plex.BettiNumbers(new int[] {1, 0, 0, 0, 21}));
189    
190          p = Plex.Persistence();
191          ds = PointData.Discrete.DSphereKskeleton(6,5);
192          gen_dk = Plex.FilterInfinite(p.computeIntervals(ds));
193          assertEquals("S(d,k) homology", gen_dk, 
194                       new Plex.BettiNumbers(new int[] {1, 0, 0, 0, 0, 7}));
195        }
196      }
