024    package edu.stanford.math.plex;
025    
026    import java.util.*;
027    
028    /**
029     * The <code>EuclideanArrayData</code> class is the simplest implementation
030     * of NSpace. Intended for smallish data sets.
031     *
032     * @version $Id: EuclideanArrayData.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
033     */
034    
035    public final class EuclideanArrayData extends PointData.NSpace {
036    
037      final double[] points;
038      final int dimensions;
039    
040      /**
041       * The number of dimensions of the space.
042       *
043       * <p>
044       *
045       * @return     the dimension of the data.
046       */
047      public final int dimension() {
048        return dimensions;
049      }
050       
051      /**
052       * The number of data points.
053       *
054       * <p>
055       *
056       * @return     the number of points
057       */
058      public final int count() {
059        if (points != null)
060          return ((points.length/dimensions)-1);
061        else
062          return 0;
063      }
064       
065      /**
066       * Euclidean distance.
067       *
068       * <p>
069       *
070       * @param      p1   the first point
071       * @param      p2   the second point
072       * @return     the Euclidean distance between p1 and p2.
073       */
074      public final double distance (int p1, int p2) {
075        if (p1 == p2)
076          return 0.0;
077        else if ((p1 == 0) || (p2 == 0))
078          return Double.MAX_VALUE;
079        else {
080          int p1_base = dimensions * p1;
081          int p2_base = dimensions * p2;
082          double accum = 0.0;
083       
084          for (int i = 0; i < dimensions; i++) {
085            double diff = points[p1_base + i] - points[p2_base + i];
086            accum += diff * diff;
087          }
088    
089          return Math.sqrt(accum);
090        }
091    
092      }
093    
094      /**
095       * Get coordinates. No checking.
096       *
097       * <p>
098       *
099       * @param      p   the point
100       * @param      i   the coordinate
101       * @return     The i-th coordinate of p;
102       */
103      public final double coordinate (int p, int i) {
104        return points[(dimensions * p) + i];
105      }
106    
107      /**
108       * Get point. No checking.
109       *
110       * <p>
111       *
112       * @param      p   the point
113       * @param      vec   place to store point
114       * @return     Coordinates of p;
115       */
116      public final double[] get_pt (int p, double[] vec) {
117        assert(dimensions == vec.length);
118        for (int i = 0; i < dimensions; i++) {
119          vec[i] = points[(dimensions * p)+i];
120        }
121        return vec;
122      }
123    
124      // Make sure that the data and dimension arguments are not absurd.
125      private boolean pointsAreConsistent(double[] data, int dimension) {
126        if (dimension < 1)
127          return false;
128        if ((data.length % dimension) != 0)
129          return false;
130        if (data.length <= dimension)
131          return false;
132        for (int i = 0; i < dimension; i++) {
133          if (data[i] != 0.0)
134            return false;
135        }
136        return true;
137      }
138    
139      // Don't call this one.
140      private EuclideanArrayData() {
141        super();
142        points = null;
143        dimensions = 0;
144      }
145    
146      // Constructor for this class.
147      public EuclideanArrayData(double[] data, int dimension) {
148        super();
149        assert(pointsAreConsistent(data, dimension));
150        dimensions = dimension;
151        points = data;
152      }
153    
154      // Constructor for test use.
155      public EuclideanArrayData(int num_pts, int dimension) {
156        super();
157        Random rand = new Random(0L);
158        double[] data = new double[(num_pts + 1) * dimension];
159        for (int i = 1; i <= num_pts; i++) {
160          for (int j = 0; j < dimension; j++) {
161            data[(i * dimension) + j] = rand.nextDouble();
162          }
163        }
164        assert(pointsAreConsistent(data, dimension));
165        dimensions = dimension;
166        points = data;
167      }
168    
169      // Constructor for test use.
170      public EuclideanArrayData(int num_pts, int dimension, double min, double max) {
171        super();
172        Random rand = new Random(0L);
173        double[] data = new double[(num_pts + 1) * dimension];
174        double len = max - min;
175        for (int i = 1; i <= num_pts; i++) {
176          for (int j = 0; j < dimension; j++) {
177            data[(i * dimension) + j] = (len * rand.nextDouble()) + min;
178          }
179        }
180        assert(pointsAreConsistent(data, dimension));
181        dimensions = dimension;
182        points = data;
183      }
184    }
