065      /**
066       * Tests some behavior.
067       *
068       * @exception  whatever
069       * @throws what is the difference between these 2
070       *
071       * @see        java.lang.System#getProperty(java.lang.String)
072       * @see        SecurityManager#checkPermission
073       */    
074      @Test
075        public void testSomeBehavior() {
076    
077        // check some misc things about RipsStreams
078        {
079          assertEquals("edge indices are okay", true, RipsStream.check_edge_index(222));
080          PointData data = (PointData) new Torus();
081          PointData rdata = (PointData) new EuclideanArrayData(40, 4);
082          RipsStream str = new RipsStream(.1, 2.0, 3, data);
083          RipsStream rstr = new RipsStream(.05, 2.0, 3, rdata);
084          RipsStream zstr = new RipsStream(0.0, 2.0, 3, rdata);
085          SimplexStream stream = (SimplexStream) str;
086          SimplexStream rstream = (SimplexStream) rstr;
087          SimplexStream zstream = (SimplexStream) zstr;
088          assertTrue("big enough stream", (rstr.size() > 100000));
089          assertTrue("big enough stream", (zstr.size() > 0));
090        }
091          
092        // check iterator and stream next for equivalence
093        {
094          Persistence p = Plex.Persistence();
095          PointData data = (PointData) new EuclideanArrayData(100, 3); 
096          double max_dist = 2.0;
097          RipsStream str = new RipsStream(.1, max_dist, 2, data);
098          RipsStream strcpy = new RipsStream(.1, max_dist, 2, data);
099          Iterator<Simplex> iterator = strcpy.iterator();
100    
101          if (str.size() < 1000000) {
102            while(iterator.hasNext()) {
103              assertTrue("iterator check", (str.hasNext() &&
104                                            (iterator.next()).equals(str.next())));
105            }
106            assertTrue("iterator check", !str.hasNext());
107          }
108        }
109    
110        // make sure that the fixed dimensional iterators work
111        {
112          PointData data = (PointData) new EuclideanArrayData(20, 3); 
113          RipsStream rc = Plex.RipsStream(.1, 20, 3, data);
114          RipsStream zc = Plex.RipsStream(0, 20, 3, data);
115          int test_dimension = 1;
116          Iterator<Simplex> it_dim = rc.iterator(test_dimension);
117          Iterator<Simplex> it = rc.iterator();
118          int counter = 0;
119          while(it.hasNext()) {
120            Simplex n = it.next();
121            if (n.dimension() == test_dimension) {
122              Simplex n_d = it_dim.next();  
123              counter++;
124              assertEquals("fixed d iterator", n, n_d);
125            }
126          }
127          assertTrue("fixed d iterator test", (counter > 0));
128          assertFalse("fixed d iterator done", it_dim.hasNext());
129          double cutoff = .75;
130          Simplex[] matches_le = rc.matchingSimplices(test_dimension, cutoff, 
131                                                      SimplexStream.ComparisonType.LE);
132          Simplex[] matches_gt = rc.matchingSimplices(test_dimension, cutoff, 
133                                                      SimplexStream.ComparisonType.GT);
134          assertTrue("matching simplex test", ((matches_gt.length > 0) &&
135                                               (matches_le.length > 0)));
136          assertTrue("matching simplex test", (counter == (matches_gt.length +
137                                                           matches_le.length)));
138          Simplex[] zmatches_le = zc.matchingSimplices(test_dimension, cutoff, 
139                                                       SimplexStream.ComparisonType.LE);
140          Simplex[] zmatches_gt = zc.matchingSimplices(test_dimension, cutoff, 
141                                                       SimplexStream.ComparisonType.GT);
142          assertTrue("matching simplex test", ((zmatches_gt.length > 0) &&
143                                               (zmatches_le.length > 0)));
144          assertTrue("matching simplex test", (counter == (zmatches_gt.length +
145                                                           zmatches_le.length)));
146        }
147    
148        // Check our calculation for the Homology of the 2-Torus for both a
149        // RipsStream and an explicitly constructed one.
150        {
151          Persistence p = Plex.Persistence();
152          int n = 20;
153          int d = 2;
154          double incr = (2.0 * PI)/n;
155          Torus tor = Plex.Torus(n, d);
156          double slop = 1.2 * sqrt(((double)d) * 
157                                   (((1.0 - cos(incr)) * (1.0 - cos(incr))) + 
158                                    (sin(incr) * sin(incr))));
159          double granularity = slop/1000.0;
160          RipsStream rc = Plex.RipsStream(granularity, slop, d+1, tor);
161          RipsStream zc = Plex.RipsStream(0,           slop, d+1, tor);
162          PersistenceInterval.Float[] rci = p.computeIntervals(rc);
163          Plex.BettiNumbers gen_rc = Plex.FilterInfinite(rci);
164          PersistenceInterval.Float[] zci = p.computeIntervals(zc);
165          Plex.BettiNumbers expected_rc = 
166            new Plex.BettiNumbers(new int[] {1, 2, 1});
167          assertEquals("2-torus betti numbers", gen_rc, expected_rc);
168          assertEquals("rci/zci lengths", rci.length, zci.length);
169    
170          if (false) {
171            System.out.printf("\n\n rci/zci (granularity = %f): \n", granularity);
172            for (int idx = 0; idx < rci.length; idx++) {
173              System.out.printf("%s/%s\n", rci[idx].toString(), zci[idx].toString());
174            }
175            System.out.printf("\n\n");
176          }
177          assertTrue("0-granularity test", Plex.equalPersistenceIntervals(rci, zci, granularity));
178    
179          if (false) {
180            // example for the doc -- make sure it works
181            System.out.printf("\n\np = Plex.Persistence()\n");
182            System.out.printf("tor = Plex.Torus(%d, %d)\n", n, d);
183            System.out.printf("rc = Plex.RipsStream(%f, %f, %d, tor)\n",
184                              slop/10, slop, d+1);
185            System.out.printf("rc.size()\n");
186            System.out.printf("res = Plex.FilterInfinite(p.computeIntervals(rc))\n");
187            System.out.printf(" %s\n\n", gen_rc.toString());
188          }
189        }
190    
191        // test the homology calculation for a handmade torus
192        {
193          SimplexStream ttor = TmpStream.Torus();
194          Persistence pt = new Persistence();
195          Plex.BettiNumbers gen_tt = 
196            Plex.FilterInfinite(pt.computeIntervals(ttor));
197          Plex.BettiNumbers expected_tt = 
198            new Plex.BettiNumbers(new int[] {1, 2, 1});
199          assertEquals("2-torus betti numbers", gen_tt, expected_tt);
200        }
201
