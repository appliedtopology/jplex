030    /**
031     * A <code>ExplicitStream</code> instance is SimplexStream whose elements
032     * are given explicitly, along with associated values of a persistence
033     * parameter. Instances of ExplicitStream can be edited after creation,
034     * unlike other SimplexStream subclasses, and an expected use is to create
035     * empty ExplicitStreams and add simplices (and associated persistence
036     * parameters) to the stream. Instances of this class exist in two states,
037     * open (meaning open for additions) and closed. A stream that is closed
038     * can be reopened at any time, but streams that are open cannot be used as
039     * arguments to the methods in the Persistence class, iterated over, or
040     * have their contents dumped for examination. Further, iterators or dump
041     * state created when an ExplicitStream is closed become invalid (and
042     * useless) once a stream is reopened. There is no explicit way to open one
043     * of these streams -- it happens implicitly when an empty stream is
044     * created, and when the stream is edited by adding/removing simplices. You
045     * can create an iterator on an open stream, but you can't use the iterator
046     * until the stream is closed. Dumping the contents of a stream will close
047     * it, as will calling the <code>close()</code> method. There is a
048     * verification process which must succeed in order for a close to happen
049     * -- if the stream is in an inconsistent state (e.g., missing faces or the
050     * persistence parameters of a face exceed that of a simplex), an
051     * IllegalStateException will be thrown. It should be possible to catch
052     * this exception, fix the problem, and proceed, but it might not work in
053     * every case. There is also a <code>verify()</code> method which will
054     * safely tell if a close() would succeed, and a verbose version that will
055     * print messages about all of the errors found.
056     *
057     * @see        edu.stanford.math.plex.RipsStream
058     *
059     * @version $Id: ExplicitStream.java,v 1.1.1.1 2008/09/12 01:53:25 hsexton Exp $
060     */
061    public final class ExplicitStream extends SimplexStream {
062    
063      // Store the simplices entries by dimension -- recall that the
064      // Persistence algorithm only requires that the stream return the
065      // simplices in persistence order within each dimension, and that
066      // all of the faces of a simplex are seen before the simplex is
067      // seen. This means that we can sort the simplices either by
068      // dimension first, or by findex first. The SimplexStream.Stack,
069      // used by the "industrial strength" streams in Plex, sorts first
070      // by findex, and then dimension. However, because we want to be
071      // able to return the contents of these streams as matrices to
072      // Matlab and R, it is more convenient to keep the simplices
073      // grouped by dimension, so that all of the simplices of a given
074      // dimension can be returned as single matrix with dimension+1
075      // columns.
076      private Simplex[][] simplices;
077      // map between persistence parameters and findices.
078      private double[] persistence_parameters;
079    
080      // False if the stream is open for the addition/removal of
081      // simplices. Closing the stream results in updating the simplices and
082      // lengths arrays, and confirming that the set of simplices does form a
083      // simplicial complex with consistent parameters.
084      private boolean closed;
085    
086      // Used to iterate over simplices if the stream is closed -- get reset
087      // every time it is open.
088      private int current_dimension;
089      private int current_index;
090      // Increment every time this is opened. Iterators for this class can only
091      // work when their cached open_count is == to the stream's open_count.
092      private int open_count;
093    
094      // The state below is used whenever the stream is open for additions, and
095      // gets flushed at the point that the stream is closed.
096    
097      // Used to map simplices being added to stream to their persistence
098      // parameters. The contents of this table are always new Simplex
099      // instances with unset _findex slots.
100      private HashMap<Simplex,Double> simplex_table;
101      // Any parameter seen goes in this table. When we are done the contents
102      // are used to fill in the persistence_parameters array, which is then
103      // sorted, and indices from there are used to fill in the simplices in
104      // the simplex_table, which are in turn stored, by dimension and then
105      // sorted by findex within the dimensions, in the simplices array.
106      private HashSet<Double> parameters_table;

110       * A DComplex (for Dimensional Complex) is the set of simplices (and
111       * associated persistence parameters) of specified dimension for an
112       * ExplicitStream. The slots are accessed only by read methods. It is
113       * expected that this will be used primarily to connect Plex with Matlab.
114       *
115       */
116      public static class DComplex {
117        private final ExplicitStream parent;
118        private final int dimension;
119        private final int open_count;
120        private final int[][] C;
121        private final double[] F;

123        /**
124         * Constructor for DComplex;
125         *
126         * <p>
127         * @param      parent The ExplicitStream from which the simplices came.
128         * @param      dimension The dimension of the simplices.
129         * @param      C The actual array of vertex arrays (simplices).
130         * @param      F The array of associated persistence parameters.
131         */
132        public DComplex(ExplicitStream parent, int dimension, int[][] C, double[] F) {
133          this.parent = parent;
134          this.open_count = parent.open_count;
135          this.dimension = dimension;
136          this.C = C;
137          this.F = F;

140        // Are the parent stream and the iterator consistent?
141        private final boolean parent_is_consistent() {
142          return (parent.closed && (open_count == parent.open_count));
143        }

145        /**
146         * Is there a larger dimensional DComplex for this stream?
147         * <p>
148         * @return true if the parent is consistent with this DComplex
149         * and there is are larger dimensional simplices in the parent, else
150         * false.
151         *
152         */
153        public boolean hasNext() {
154          if (!parent_is_consistent())
155            return false;
156          else 
157            return (dimension < parent.maxDimension());
158        }

161         * Return a DComplex of the next largest dimension, if there is one,

163         *
164         * <p>
165         * @return A DComplex of dimension 1 more, if there is one, else null.
166         */
167        public DComplex next() {
168          if (!parent_is_consistent())
169            return null;
170          else 
171            return parent.dump(dimension+1);
172        }

219      /**
220       * Return a DComplex of the requested dimension, if there is one, else null;
221       *
222       * <p>
223       * @param      dim The dimension of the simplices we want.
224       * @return     Return a DComplex if dim >= 0 and <= maxDimension(), else null.
225       *
226       */
227      public DComplex dump(int dim) {
228        if (!closed)
229          close();
230        if ((dim < 0) || (dim > maxDimension()))
231          return null;
232        else {
233          int[][] cvec = new int[simplices[dim].length][];
234          double[] fvec = new double[cvec.length];
235          for (int i = 0; i < cvec.length; i++) {
236            Simplex s = simplices[dim][i];
237            cvec[i] = s.vertices();
238            fvec[i] = persistence_parameters[s.findex()];
239          }
240          return new DComplex(this, dim, cvec, fvec);
241        }
242      }

244      /**
245       * Is there a next Simplex in the stream?
246       *
247       * <p>
248       * @return     Return true if the stream is not yet empty.
249       *
250       */
251      public final boolean hasNext() {
252        // NOTE: For this to work it must be the case that any CLOSED
253        // ExplicitStream has the property that the Simplex[] for the highest
254        // dimension has positive length. That is, if the largest dimensional
255        // Simplex in the stream has dimension 3, then simplices.length must
256        // be 4. It is, of course, possible that simplices[i] has 0 length
257        // for some i, just not for i == simplices.length-1.
258        if (!closed)
259          return false;
260        else 
261          return ((current_dimension < (simplices.length-1)) ||
262                  (current_dimension == (simplices.length-1)) &&
263                  (current_index < simplices[simplices.length-1].length-1));
264      }
265    
266      // Go to the "next" index pair. This code assumes that there is such a
267      // pair, and so needs to be called only when we know that this is the
268      // case.
269      private final void skip_to_next_index_pair() {
270        current_index++;
271        while(current_index >= simplices[current_dimension].length) {
272          current_index = 0;
273          current_dimension++;
274        }
275      }
276      
277      /**
278       * Next Simplex in the stream.
279       *
280       * <p>
281       * @return     Return the smallest remaining Simplex instance.
282       *
283       */
284      public Simplex next() {
285        if (!hasNext())
286          return null;
287        Simplex next = simplices[current_dimension][current_index];
288        skip_to_next_index_pair();
289        return next;
290      }

337       * Constructor for ExplicitStream.
338       *
339       * <p>
340       * Make an empty, open, ExplicitStream.
341       */
342      public ExplicitStream() {
343        simplices = null;
344        persistence_parameters = null;
345        closed = false;
346        current_dimension = 0;
347        current_index = 0;
348        open_count = 1;
349        parameters_table = new HashSet<Double>(100);
350        simplex_table = new HashMap<Simplex,Double>(100);
351      }

353      // Open the stream for further additions, after creating the internal
354      // state needed to record these additions. In particular, we put the
355      // existing state into these tables and then throw away that existing
356      // state.
357      private void open() {
358        if (closed) {
359          if (parameters_table == null)
360            parameters_table = new HashSet<Double>(100);
361          if (simplex_table == null)
362            simplex_table = new HashMap<Simplex,Double>(100);
363          if (simplices != null) {
364            for(int d = 0; d < simplices.length; d++) {
365              for(int i = 0; i < simplices[d].length; i++) {
366                Simplex s = simplices[d][i];
367                Double P = new Double(persistence_parameters[s.findex()]);
368                simplex_table.put(Simplex.getSimplex(s.vertices()), P);
369                parameters_table.add(P);
370              }
371            }
372          }
373          simplices = null;
374          persistence_parameters = null;
375          closed = false;
376          open_count++;
377        }
378      }

380      /**
381       * Convert a SimplexStream into an explicit one.
382       *
383       * <p>
384       * @param      str A SimplexStream instance.
385       * @return     An ExplicitStream with equivalent Simplex entries.
386       *
387       */
388      public static ExplicitStream makeExplicit(SimplexStream str) {
389        if (str instanceof ExplicitStream)
390          return (ExplicitStream) str;
391        else {
392          ExplicitStream estr = new ExplicitStream();
393          estr.open();
394          if (str instanceof Iterable) {
395            Iterator<Simplex> iter = str.iterator();
396            while(iter.hasNext()) {
397              Simplex s = iter.next();
398              estr.add(s.vertices(), str.convert_filtration_index(s.findex()));
399            }
400          } else {
401            while(str.hasNext()) {
402              Simplex s = str.next();
403              estr.add(s.vertices(), str.convert_filtration_index(s.findex()));
404            }
405          }
406          estr.close();
407          return estr;
408        }
409      }

459      // Internal version -- add a Simplex with a specified persistence parameter.
460      private final void add(Simplex s, double p) {
461        if (closed)
462          open();
463        Double P = new Double(p);
464        parameters_table.add(P);
465        Double sP = simplex_table.get(s);
466        if (sP == null) {
467          simplex_table.put(s, P);
468        } else {
469          simplex_table.put(s, (P.doubleValue() > sP.doubleValue())?sP:P);
470        }
471      }

530      // Make sure that all of the faces are present of a Simplex are
531      // present. If some are missing, add them with the smallest persistence
532      // parameter associated with that face in the pending table.
533      private void ensure_faces(Simplex s, Double P, HashMap<Simplex,Double> pending) {
534        assert(P instanceof Double);
535        if (s.dimension() > 0) {
536          SimplexFaceIterator fi = new SimplexFaceIterator(s, s.dimension()-1);
537          // add missing faces
538          while (fi.hasNext()) {
539            Simplex face = Simplex.getSimplex(fi.next());
540            if (!simplex_table.containsKey(face)) {
541              if (pending.containsKey(face)) {
542                Double prior_faceP = pending.get(face);
543                if (prior_faceP.doubleValue() < P.doubleValue())
544                  P = prior_faceP;
545              }
546              pending.put(Simplex.getSimplex(face.vertices()), P);
547              ensure_faces(face, P, pending);
548            }
549          } 
550        }
551      }
552    
553      /**
554       * Make sure all of the faces of simplices in simplex_table are present
555       * in the table as well.  <p> In doing so we use the smallest relevant
556       * persistence parameter. This does NOT ensure that the full complex will
557       * be consistent, but if it fails, the complex was doomed anyway.  <p>
558       *
559       */
560      public void ensure_all_faces() {
561        if (!closed) {
562          HashMap<Simplex,Double> pending = new HashMap<Simplex,Double>(100);
563          for(Map.Entry<Simplex,Double> E : simplex_table.entrySet()) {
564            Simplex s = E.getKey();
565            Double P = E.getValue();
566            ensure_faces(s, P, pending);
567          }
568          for(Map.Entry<Simplex,Double> E : pending.entrySet()) {
569            Simplex s = E.getKey();
570            Double P = E.getValue();
571            parameters_table.add(P);
572            simplex_table.put(s, P);
573          }
574        }
575      }

577      // Realloc() for int[].
578      private final int[] ensure_length(int[] v, int index) {
579        if (index < v.length)
580          return v;
581        int[] new_v = new int[index+1];
582        for (int i = 0; i < v.length; i++) 
583          new_v[i] = v[i];
584        return new_v;
585      }

587      // Iterate over all of the faces of s and make sure that they are present
588      // in simplex_table. This only applies to open ExplicitStreams, since the
589      // process of closing a stream requires that it be consistent.
590      private boolean faces_are_consistent(Simplex s, Double P,
591                                           boolean error, boolean print) {
592        boolean success = true;
593        if (!closed) {
594          SimplexFaceIterator fi = new SimplexFaceIterator(s, s.dimension()-1);
595          // check that the faces are all present and have smaller persistence parameter
596          while (fi.hasNext()) {
597            Simplex face = Simplex.getSimplex(fi.next());
598            Double faceP = simplex_table.get(face);
599            if (faceP == null) {
600              success = false;
601              if (error) {
602                persistence_parameters = null;
603                throw new
604                  IllegalStateException("Simplex " + 
605                                        s.toString() + 
606                                        " is in the stream, but not face " +
607                                        face.toString());
608              } else if (print) {
609                System.out.printf("Simplex %s is present, but its face %s is not.\n",
610                                  s.toString(), face.toString());
611              }
612            } else if ((faceP.doubleValue() > P.doubleValue()) ||
613                       (face.findex() > s.findex())) {
614              success = false;
615              if (error) {
616                persistence_parameters = null;
617                throw new
618                  IllegalStateException("Simplex " + 
619                                        s.toString() + 
620                                        " has persistence value " +
621                                        P.toString() +
622                                        " filtration index " +
623                                        s.findex() +
624                                        ", but persistence value of " +
625                                        face.toString() +
626                                        " or filtration index of " +
627                                        face.findex() +
628                                        " is larger");
629              } else if (print) {
630                System.out.printf
631                  ("Simplex %s has value %.4f, but face %s has value %.4f.\n",
632                   s.toString(), P.doubleValue(), 
633                   face.toString(), faceP.doubleValue());
634              }
635            }
636          }
637        }
638        return success;
639      }

641      /**
642       * Quietly verify the ExplicitStream. Any successfully closed stream must
643       * be consistent.  <p>
644       * @return     Return true if the stream is consistent, else false.
645       *
646       */
647      public boolean verify() {
648        return verify(false);
649      }

651      /**
652       * Verify the ExplicitStream. This is only applied to a stream that has been
653       * opened for additions. Any successfully closed stream must be consistent.
654       *
655       * <p>
656       * @param      verbose if true print a message for every mistake found.
657       * @return     Return true if the stream is consistent, else false.
658       *
659       */
660      public boolean verify(boolean verbose) {
661        boolean success = true;
662        if (!closed) {
663          for(Map.Entry<Simplex,Double> E : simplex_table.entrySet()) {
664            Simplex s = E.getKey();
665            Double P = E.getValue();
666            assert(P instanceof Double);
667            boolean good_face = faces_are_consistent(s, P, false, verbose);
668            if (success)
669              success = good_face;
670          }
671        }
672        return success;
673      }

675      /**
676       * Close the stream the current additions, provided that the contents are
677       * consistent and complete. Otherwise, throw an error indicating the
678       * problem. Note that even if the close is successful, further additions
679       * can be made at any time. It isn't obvious that this method needs to be
680       * public, but it makes some testing simpler.
681       */
682      public void close() {
683        if (!closed) {
684          int pcount =
685            (parameters_table != null)?parameters_table.size():0;
686          persistence_parameters = new double[pcount];
687          if (parameters_table != null) {
688            int pcounter = 0;
689            for (Double P : parameters_table) {
690              persistence_parameters[pcounter++] = P.doubleValue();
691            }
692          }
693          Arrays.sort(persistence_parameters);
694          if (simplex_table != null) {
695            int[] dimensions = new int[10];
696            int max_dimension = -1;
697            for(Map.Entry<Simplex,Double> E : simplex_table.entrySet()) {
698              Simplex s = E.getKey();
699              Double P = E.getValue();
700              assert(P instanceof Double);
701              dimensions = ensure_length(dimensions, s.dimension());
702              dimensions[s.dimension()]++;
703              max_dimension = Math.max(max_dimension, s.dimension());          
704              // check that the faces are all present and have persistence
705              // parameters that are no larger
706              faces_are_consistent(s, P, true, false);
707            }
708            simplices = new Simplex[max_dimension+1][];
709            int[] counters = new int[max_dimension+1];
710            for (int i = 0; i < simplices.length; i++) 
711              simplices[i] = new Simplex[dimensions[i]];
712            for(Map.Entry<Simplex,Double> E : simplex_table.entrySet()) {
713              Simplex s = E.getKey();
714              Double P = E.getValue();
715              double s_p = P.doubleValue();
716              if (s.findex() < 0)
717                // only bother setting the findex values if they haven't been set
718                s.setfindex(Arrays.binarySearch(persistence_parameters, s_p));
719              simplices[s.dimension()][counters[s.dimension()]++] = s;
720            }
721            // Sort the simplices entries within each dimension. This will
722            // result in a state that allows us to return the contents of the
723            // stream in an order suitable for the Persistence calculation.
724            for (int i = 0; i < simplices.length; i++) {
725              // assert that the simplices subarrays are full
726              assert (simplices[i].length == counters[i]);
727              Simplex.persistence_sort(simplices[i]);
728            }
729          }
730          simplex_table = null;
731          parameters_table = null;
732          closed = true;
733        }
734      }
735    
736      /**
737       * How many simplices are in the stream?
738       *
739       * <p>
740       * @return The number of simplices in a closed stream, else 0 if the
741       * stream is open.
742       *
743       */
744      public final int size() {
745        if (!closed)
746          return 0;
747        else {
748          if (simplices == null)
749            return 0;
750          else {
751            int total = 0;
752            for (int i = 0; i < simplices.length; i++) {
753              total += simplices[i].length;
754            }
755            return total;
756          }
757        }
758      }
759    
760      /**
761       * Max dimension of simplices in the stream.
762       *
763       * <p>
764       * @return Max dimension of simplices in the stream, if closed and
765       * non-empty, else -1.
766       *
767       */
768      public final int maxDimension() {
769        if (!closed)
770          return -1;
771        else {
772          if (simplices == null)
773            return -1;
774          else {
775            return simplices.length-1;
776          }
777        }
778      }

805      /**
806       * Make an iterator for the stream elements of fixed dimension.
807       * <p>
808       *
809       * @param      d dimension of simplices of interest.
810       * @return  Iterator<Simplex> instance for the stream.
811       *
812       * @see        java.util.Iterator
813       */

815        return new ExplicitStreamIterator(this, d);
816      }
817    
818      /**
819       * Instances provide Iterator<Simplex> for non-destructive iterating
820       * over entries. That is, repeated iteration over an ExplicitStream
821       * instance is possible with multiple instances of this class.
822       */
823      private static final class ExplicitStreamIterator implements Iterator<Simplex> {
824        private int current_index;
825        private int current_dimension;
826        private final boolean fixed_dimension;
827        private final int open_count;
828        private final ExplicitStream parent;
829    
830        private ExplicitStreamIterator(ExplicitStream str) {
831          current_dimension = 0;
832          current_index = 0;
833          open_count = str.open_count;
834          fixed_dimension = false;
835          parent = str;
836        }
837    
838        private ExplicitStreamIterator(ExplicitStream str, int dimension) {
839          if ((dimension < 0) || (dimension >= str.simplices.length))
840            throw new IllegalArgumentException(dimension + 
841                                               " must be >= 0 and <= " + 
842                                               (str.simplices.length-1)); 
843          current_dimension = dimension;
844          current_index = 0;
845          open_count = str.open_count;
846          fixed_dimension = true;
847          parent = str;
848        }
849    
850        // Go to the "next" index pair. This code assumes that there is such a
851        // pair, and so needs to be called only when we know that this is the
852        // case.
853        private final void skip_to_next_index_pair() {
854          current_index++;
855          while(current_index >= parent.simplices[current_dimension].length) {
856            current_index = 0;
857            current_dimension++;
858          }
859        }
860    
861        // Are the parent stream and the iterator consistent?
862        private final boolean parent_is_consistent() {
863          return (parent.closed && (open_count == parent.open_count));
864        }
865          
866        // NOTE: For this to work it must be the case that any CLOSED
867        // ExplicitStream has the property that the Simplex[] for the highest
868        // dimension has positive length. That is, if the largest dimensional
869        // Simplex in the stream has dimension 3, then simplices.length must
870        // be 4. It is, of course, possible that simplices[i] has 0 length
871        // for some i, just not for i == simplices.length-1.
872        private final boolean has_next_index() {
873          return (fixed_dimension?
874                  (current_index < parent.simplices[current_dimension].length):
875                  ((current_dimension < (parent.simplices.length-1)) ||
876                   (current_dimension == (parent.simplices.length-1)) &&
877                   (current_index < 
878                    parent.simplices[parent.simplices.length-1].length-1)));
879        }
880    
881        /**
882         * Returns <tt>true</tt> if the iterator has more simplices.
883         *
884         * @return <tt>true</tt> if the iterator has more simplices, else
885         * <tt>false</tt>.
886         */
887        public final boolean hasNext() {
888          if (!parent_is_consistent())
889            return false;
890          else
891            return has_next_index();
892        }
893    
894        /**
895         * Returns the next Simplex in the iteration of the stream.
896         * Used in concert with the {@link #hasNext()} method returns
897         * return each Simplex in the stream exactly once. Does
898         * <em>not</em> have any side effects on the stream.
899         * <p>
900         * @return the next Simplex in the stream.
901         * @exception NoSuchElementException stream has no more elements.
902         */
903        public final Simplex next() {
904          if (!parent_is_consistent())
905            throw new
906              IllegalStateException("The parent stream (" + parent.toString() +
907                                    ") has changed since this iterator was created.");
908          if (!has_next_index())
909            throw new NoSuchElementException();
910              
911          Simplex next = parent.simplices[current_dimension][current_index];
912          skip_to_next_index_pair();
913          return next;
914        }
915    
916        /**
917         * Unsupported remove() operation.
918         *
919         * @exception UnsupportedOperationException 
920         */
921        public void remove() {
922          throw new UnsupportedOperationException();
923        }
924      }
925    
926        /**
927         * Convert a Simplex[][] into an ExplicitStream.
928         * <p>
929         * @param complex A Simplex[][] collection of simplices which are
930         * believed to a simplicial complex with findex values usable as
931         * persistence parameters.
932         * @return An ExplicitStream instance the encloses the Simplex instances.
933         */
934      public static ExplicitStream enclose (Simplex[][] complex) {
935        ExplicitStream estr = new ExplicitStream();
936        estr.open();
937        for (int i = 0; i < complex.length; i++) {
938          Simplex[] sarray = complex[i];
939          for (int j = 0; j < sarray.length; j++) {
940            Simplex s = sarray[j];
941            double p = (s.findex() < 0)?0.0:((double)s.findex());
942            estr.add(s.copy(), p);
943          }
944        }
945        estr.close();
946        return estr;
947      }
