062        int[] test = new int[len];

067        }

072      public static class TestComp extends PlexSort.Comp {
073        public final boolean increasing;
074        public int fn(int i, int j) { 
075          if (increasing) 
076            return (i - j); 
077          else  
078            return (j - i);
079        }
080        
081        public TestComp(boolean incr) {
082          increasing = incr;
083        }
084    
085        public boolean check_random_vec(int[] vec) {
086          for (int i = 1; i < vec.length; i++) {
087            if (increasing) {
088              if (vec[i-1] > vec[i])
089                return false;
090            } else {
091              if (vec[i-1] < vec[i])
092                return false;
093            }
094          }
095          return true;
096        }
097    
098      }
099    
100        public static boolean check_vec(int[] vec) {
101          for (int i = 1; i < vec.length; i++) {
102            if (vec[i-1] > vec[i]) {
103              System.out.printf("\n### ");
104              for (int j = 0; j < vec.length; j++) {
105                if (j == (i-1))
106                  System.out.printf(" >>> %d ", vec[j]);
107                else if (j == i)
108                  System.out.printf("%d <<< ", vec[j]);
109                else 
110                  System.out.printf("%d ", vec[j]);
111              }
112              System.out.printf("\n\n");
113              return false;
114            }
115          }
116          return true;
117        }
