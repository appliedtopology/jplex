036      /**
037       * Instances of the <code>Comp</code> inner class are used as comparison
038       * closures for the sorting routine <code>comp_sort</code>. In addition
039       * to the obvious uses, such as sorting in decreasing order, using
040       * closures allows us to sort arrays of points by increasing or
041       * decreasing density (or any other real-valued function on the points).
042       * There are also some slightly non-obvious applications, which we
043       * probably won't use here, such a computing the inverse of a permutation
044       * stored in an array p[]. If p is [1,2,0] -- that is, the permutation
045       * mapping 0->1, 1->2, 2->0, then let x = [0,1,2], and sort the entries
046       * of x[i] so that x[i] is less than x[j] iff p[x[i]] < p[x[j]]. Then the
047       * sorted contents of x are [2,0,1]. This is the inverse of p[].
048       *
049       */
050      public static abstract class Comp {
051        /**
052         * Compare two int arguments, and return an integer less than, equal
053         * to, or greater than 0, depending on whether i is respectively less,
054         * equivalent, or greater than, j.
055         *
056         * @param      i The first argument.
057         * @param      j The second argument.
058         * @return     an int that is -/0/+ -- intuitively, return (i-j) for 
059         *             an increasing sort.
060         */
061        public abstract int fn(int i, int j);
062      }
063    
064      public static abstract class CompL {
065        /**
066         * Compare two long arguments, and return an integer less than, equal
067         * to, or greater than 0, depending on whether i is respectively less,
068         * equivalent, or greater than, j.
069         *
070         * @param      i The first argument.
071         * @param      j The second argument.
072         * @return     an int that is -/0/+ -- intuitively, return (i-j) for 
073         *             an increasing sort.
074         */
075        public abstract int fn(long i, long j);
076      }

078      public static abstract class CompObj {
079        /**
080         * Compare two object arguments, and return an integer less than, equal
081         * to, or greater than 0, depending on whether i is respectively less,
082         * equivalent, or greater than, j.
083         *
084         * @param      o1 The first argument.
085         * @param      o2 The second argument.

087         *             an increasing sort.
088         */
089        public abstract int fn(Object o1 , Object o2);
090      }

092      // Exchange contents of x[i] and x[j].
093      private static final void swap(int[] x, int i, int j) {
094        int tmp = x[i];
095        x[i] = x[j];
096        x[j] = tmp;
097      }
098    
099      // Returns the index of the median of the three argument entries.
100      private static final int med3(int[] x, int a, int b, int c, Comp comp) {
101        int xa = x[a];
102        int xb = x[b];
103        int xc = x[c];
104        return ((comp.fn(xa, xb) < 0)?
105                ((comp.fn(xb, xc) < 0)?b:((comp.fn(xa, xc)<0)?c:a)):
106                ((comp.fn(xb, xc) > 0)?b:((comp.fn(xa, xc)>0)?c:a)));
107      }
108    
109      // Exchange subintervals [i, i+(len-1)] and [j, j+(len-1)] in x[]; 
110      private static final void swapfunc(int[] x, int i, int j, int len) {
111        int stop = j + len;
112        while (j < stop) {
113          int tmp = x[i];
114          x[i] = x[j];
115          x[j] = tmp;
116          i++;
117          j++;
118        }
119      }

179      /**
180       * Sort a subarray of an int[] using a comparison function. The sort is
181       * not stable, but it uses no extra storage, and seems to be faster in
182       * practice then even insertion sort, although it doesn't have that
183       * algorithm's guarantee of optimal worst-case behavior. <p>
184       *
185       * This code is copied with trivial alterations from "Engineering a Sort
186       * Function", by Bentley and McIlroy. I have no idea why there isn't
187       * something like this already available in the class java.util.Arrays --
188       * apparently the sort routines in that class are based on the same
189       * paper, but support for a comparison function has been removed.
190       *
191       * @param      x int[] to sort.
192       * @param      start Beginning of subarray to sort.
193       * @param      len Length of the subarray of x to sort.
194       * @param      comp PlexSort.Comp instance to use in sorting.
195       *
196       */
197      public static final void comp_sort(int[] x, int start, int len, Comp comp) {
198    
199        if (len < 7) {
200          // Use insertion sort on the smallest arrays.
201                for (int i = start; i< len + start; i++)
202            for (int j = i; (j > start) && (comp.fn(x[j-1], x[j]) > 0); j--) {
203              swap(x, j, j-1);
204            }
205                return;
206        }
207    
208        // Partition on the middle entry for small arrays.
209        int pm = start + len/2;       
210        if (len > 7) {
211                int p_bot = start;
212                int p_top = start + len - 1;
213                if (len > 40) {     
214            // For big arrays, use the pseudomedian of 9 entries.
215            int psm_incr = len/8;
216            p_bot = med3(x, p_bot, p_bot+psm_incr, p_bot+(2*psm_incr), comp);
217            pm = med3(x, pm-psm_incr, pm, pm+psm_incr, comp);
218            p_top = med3(x, p_top-(2*psm_incr), p_top-psm_incr, p_top, comp);
219                }
220          // For mid-size arrays, use the pseudomedian of 3 entries.
221                pm = med3(x, p_bot, pm, p_top, comp); 
222        }
223        int partition_value = x[pm];
224    
225        int pa = start, pb = pa;
226        int pc = start + len - 1, pd = pc;
227        while(true) {
228          int comp_val;
229                while ((pb <= pc) && 
230                 ((comp_val = comp.fn(x[pb], partition_value)) <= 0)) {
231            if (comp_val == 0) {
232              swap(x, pa, pb);
233              pa++;
234            }
235            pb++;
236                }
237                while ((pc >= pb) && 
238                 ((comp_val = comp.fn(x[pc], partition_value)) >= 0)) {
239            if (comp_val == 0) {
240              swap(x, pc, pd);
241              pd--;
242            }
243            pc--;
244                }
245          if (pb > pc)
246            break;
247                swap(x, pb, pc);
248          pb++;
249          pc--;
250        }
251    
252        int pn = start + len;
253        int s;
254    
255        s = Math.min(pa-start, pb-pa);  
256        swapfunc(x, start, pb-s, s);
257    
258        s = Math.min(pd-pc, pn-pd-1);  
259        swapfunc(x, pb, pn-s, s);
260    
261        if ((s = pb-pa) > 1)
262                comp_sort(x, start, s, comp);
263        if ((s = pd-pc) > 1)
264                comp_sort(x, pn-s, s, comp);
265      }

413      /**
414       * Take an array of indices and sort them by the values in an array.
415       * <p>
416       * @param      indices An int[] of indices to be sorted.
417       * @param      start The first entry in indices to be sort.
418       * @param      end  The first entry in indices to not be sorted.
419       * @param      vals An array of integer values against which to sort indices.
420       * @param reverse A boolean which, if true, causes the indices to be
421       * sorted into decreasing order.
422       */
423      public static void sort(int[] indices, int start, int end, 
424                              final int[] vals, boolean reverse) {
425        if (reverse) {
426          comp_sort(indices, start, end-start, 
427                    new Comp() {
428                      public int fn(int i, int j) { 
429                        return (vals[j]-vals[i]);
430                      }
431                    } );
432        } else {
433          comp_sort(indices, start, end-start, 
434                    new Comp() {
435                      public int fn(int i, int j) { 
436                        return (vals[i]-vals[j]);
437                      }
438                    } );
439        }
440      }
441    
442      public static void sort(int[] indices, int start, int end, 
443                              final double[] vals, boolean reverse) {
444        if (reverse) {
445          comp_sort(indices, start, end-start, 
446                    new Comp() {
447                      public int fn(int i, int j) { 
448                        if (vals[i] > vals[j])
449                          return -1;
450                        else if (vals[i] < vals[j])
451                          return 1;
452                        else
453                          return 0; 
454                      } 
455                    } );
456        } else {
457          comp_sort(indices, start, end-start, 
458                    new Comp() {
459                      public int fn(int i, int j) { 
460                        if (vals[i] < vals[j])
461                          return -1;
462                        else if (vals[i] > vals[j])
463                          return 1;
464                        else
465                          return 0;
466                      }
467                    } );
468        }
469      }
